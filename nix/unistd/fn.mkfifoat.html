pub fn mkfifoat<Fd: AsFd, P: ?Sized + NixPath>(

    path: &P,

Expand description

Creates new FIFO special file (named pipe) with access rights set to mode in the path specified by dirfd and path.

ยงExamples

Create a FIFO in the current working directory:

use nix::fcntl::AT_FDCWD;
use nix::unistd::mkfifoat;
use nix::sys::stat::Mode;

mkfifoat(AT_FDCWD, "fifo", Mode::S_IRWXU).unwrap();
