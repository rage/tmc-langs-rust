19pub(crate) struct InputBuffer<'a> {
20    pub bits: BitsBuffer,
21    pub buffer: &'a [u8],

25impl<'a> InputBuffer<'a> {
26    pub fn new(bits: BitsBuffer, buffer: &'a [u8]) -> Self {
27        Self {
28            bits,
29            buffer,
30            read_bytes: 0,
31        }
32    }
33
34    pub fn available_bits(&self) -> i32 {
35        self.bits.bits_in_buffer
36    }
37
38    pub fn available_bytes(&self) -> usize {
39        self.buffer.len() + (self.bits.bits_in_buffer / 4) as usize
40    }
41
42    pub fn ensure_bits_available(&mut self, count: i32) -> bool {
43        debug_assert!(0 < count && count <= 16, "count is invalid.");
44
45        // manual inlining to improve perf
46        if self.bits.bits_in_buffer < count {
47            if self.needs_input() {
48                return false;
49            }
50
51            // insert a byte to bitbuffer
52            self.bits.bit_buffer |= (self.buffer[0] as u32) << self.bits.bits_in_buffer;
53            self.advance(1);
54            self.bits.bits_in_buffer += 8;
55
56            if self.bits.bits_in_buffer < count {
57                if self.needs_input() {
58                    return false;
59                }
60                // insert a byte to bitbuffer
61                self.bits.bit_buffer |= (self.buffer[0] as u32) << self.bits.bits_in_buffer;
62                self.advance(1);
63                self.bits.bits_in_buffer += 8;
64            }
65        }
66
67        true
68    }
69
70    pub fn try_load_16bits(&mut self) -> u32 {
71        if self.bits.bits_in_buffer < 8 {
72            if self.buffer.len() > 1 {
73                self.bits.bit_buffer |= (self.buffer[0] as u32) << self.bits.bits_in_buffer;
74                self.bits.bit_buffer |= (self.buffer[1] as u32) << (self.bits.bits_in_buffer + 8);
75                self.advance(2);
76                self.bits.bits_in_buffer += 16;
77            } else if !self.buffer.is_empty() {
78                self.bits.bit_buffer |= (self.buffer[0] as u32) << self.bits.bits_in_buffer;
79                self.advance(1);
80                self.bits.bits_in_buffer += 8;
81            }
82        } else if self.bits.bits_in_buffer < 16 && !self.buffer.is_empty() {
83            self.bits.bit_buffer |= (self.buffer[0] as u32) << self.bits.bits_in_buffer;
84            self.advance(1);
85            self.bits.bits_in_buffer += 8;
86        }
87
88        self.bits.bit_buffer
89    }
90
91    fn get_bit_mask(&self, count: i32) -> u32 {
92        (1 << count) - 1
93    }
94
95    pub fn get_bits(&mut self, count: i32) -> Result<u16, InternalErr> {
96        debug_assert!(0 < count && count <= 16, "count is invalid.");
97
98        if !self.ensure_bits_available(count) {
99            return Err(InternalErr::DataNeeded);
100        }
101
102        let result = (self.bits.bit_buffer & self.get_bit_mask(count)) as u16;
103        self.bits.bit_buffer >>= count;
104        self.bits.bits_in_buffer -= count;
105        Ok(result)
106    }
107
108    pub fn copy_to(&mut self, mut output: &mut [u8]) -> usize {
109        debug_assert!(self.bits.bits_in_buffer % 8 == 0);
110
111        // Copy the bytes in bitBuffer first.
112        let mut bytes_from_bit_buffer = 0;
113        while self.bits.bits_in_buffer > 0 && !output.is_empty() {
114            output[0] = self.bits.bit_buffer as u8;
115            output = &mut output[1..];
116            self.bits.bit_buffer >>= 8;
117            self.bits.bits_in_buffer -= 8;
118            bytes_from_bit_buffer += 1;
119        }
120
121        if output.is_empty() {
122            return bytes_from_bit_buffer;
123        }
124
125        let length = min(output.len(), self.buffer.len());
126        output[..length].copy_from_slice(&self.buffer[..length]);
127        self.advance(length);
128        bytes_from_bit_buffer + length
129    }
130
131    pub fn needs_input(&self) -> bool {
132        self.buffer.is_empty()
133    }
134
135    /// <summary>Skip n bits in the buffer.</summary>
136    pub fn skip_bits(&mut self, n: i32) {
137        debug_assert!(
138            self.bits.bits_in_buffer >= n,
139            "No enough bits in the buffer, Did you call ensure_bits_available?"
140        );
141        self.bits.bit_buffer >>= n;
142        self.bits.bits_in_buffer -= n;
143    }
144
145    /// <summary>Skips to the next byte boundary.</summary>
146    pub fn skip_to_byte_boundary(&mut self) {
147        self.bits.bit_buffer >>= self.bits.bits_in_buffer % 8;
148        self.bits.bits_in_buffer -= self.bits.bits_in_buffer % 8;
149    }
150
151    fn advance(&mut self, buf: usize) {
152        self.buffer = &self.buffer[buf..];
153        self.read_bytes += buf;
154    }
