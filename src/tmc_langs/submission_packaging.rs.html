25/// Prepares a submission for further processing.
26/// Returns the sandbox image to be used for the submission.
27pub fn prepare_submission(
28    submission: PrepareSubmission,
29    target_path: &Path,
30    no_archive_prefix: bool,
31    tmc_params: TmcParams,
32    stub_clone_path: &Path,
33    stub_archive: Option<(&Path, Compression)>,
34    output_format: Compression,
35) -> Result<String, LangsError> {
36    // FIXME: workaround for unknown issues when prepare_submission is ran multiple times in parallel
37    let _m = MUTEX.lock().map_err(|_| LangsError::MutexError)?;
38    log::debug!("preparing submission for {}", submission.archive.display());
39
40    let plugin = PluginType::from_exercise(stub_clone_path)?;
41
42    let extract_dest = tempfile::tempdir().map_err(LangsError::TempDir)?;
43    let extract_dest_path = extract_dest.path().to_path_buf();
44
45    // extract base
46    log::debug!("extracting stub");
47    let ignore_list = [
48        ".DS_Store",
49        "desktop.ini",
50        "Thumbs.db",
51        ".directory",
52        "__MACOSX",
53        file_util::LOCK_FILE_NAME,
54    ];
55    if let Some((stub_zip, compression)) = stub_archive {
56        // This branch is used when a student downloads their old submission, and we take the files from the stub (the exercise template) instead of the clone path. This makes sure they cannot see the hidden tests in the downloaded file.
57        extract_with_filter(
58            plugin,
59            stub_zip,
60            compression,
61            |path| {
62                path.components().any(|c| {
63                    c.as_os_str()
64                        .to_str()
65                        .map(|s| ignore_list.contains(&s))
66                        .unwrap_or_default()
67                })
68            },
69            &extract_dest_path,
70            false,
71        )?;
72    } else {
73        // This code branch is used when we package a submission for the sandbox. We use the clone path because it contains the hidden tests, and we want the sandbox to run them.
74        for entry in WalkDir::new(stub_clone_path).min_depth(1) {
75            let entry = entry?;
76
77            if entry.path().components().any(|c| {
78                c.as_os_str()
79                    .to_str()
80                    .map(|s| ignore_list.contains(&s))
81                    .unwrap_or_default()
82            }) {
83                // path component on ignore list
84                continue;
85            }
86
87            let relative_path = entry
88                .path()
89                .strip_prefix(stub_clone_path)
90                .expect("entry is in stub clone path");
91            let target_path = extract_dest_path.join(relative_path);
92            if entry.path().is_file() {
93                file_util::copy(entry.path(), target_path)?;
94            } else if entry.path().is_dir() {
95                file_util::create_dir(target_path)?;
96            }
97        }
98    }
99
100    // extract student files from submission over base
101    log::debug!("extracting student files");
102    let file = file_util::open_file(submission.archive)?;
103    if submission.extract_naively {
104        // If the stub contains a .tmcproject.yml, preserve it so the student's cannot override it
105        let tmcproject_path = extract_dest_path.join(".tmcproject.yml");
106        let preserved_tmcproject = if tmcproject_path.exists() {
107            Some(
108                std::fs::read(&tmcproject_path)
109                    .map_err(|e| FileError::FileRead(tmcproject_path.clone(), e))?,
110            )
111        } else {
112            None
113        };
114
115        extract_project_overwrite(file, &extract_dest_path, submission.compression)?;
116
117        if let Some(bytes) = preserved_tmcproject {
118            // restore the stub's .tmcproject.yml
119            file_util::write_to_file(&bytes, &tmcproject_path)?;
120        }
121    } else {
122        // This code branch is used when we package a submission for the sandbox. This extraction method makes sure we don't allow the student to update the files they are not allowed to edit.
123        plugin.extract_student_files(file, submission.compression, &extract_dest_path)?;
124    }
125
126    // extract ide files
127    log::debug!("extracting ide files");
128    let ide_files = [
129        // netbeans
130        "nbproject",
131        // eclipse
132        ".classpath",
133        ".project",
134        ".settings",
135        // idea
136        ".idea",
137    ];
138    extract_with_filter(
139        plugin,
140        submission.archive,
141        submission.compression,
142        |path| {
143            path.components().all(|c| {
144                c.as_os_str()
145                    .to_str()
146                    .map(|s| !ide_files.contains(&s))
147                    .unwrap_or_default()
148            })
149        },
150        &extract_dest_path,
151        submission.extract_naively,
152    )?;
153
154    // write tmc params
155    if tmc_params.0.is_empty() {
156        log::debug!("no tmc params to write");
157    } else {
158        log::debug!("writing .tmcparams");
159        let tmc_params_path = extract_dest_path.join(".tmcparams");
160        let mut tmc_params_file = file_util::create_file(&tmc_params_path)?;
161        for (key, value) in tmc_params.0 {
162            // todo handle arrays, shell escapes
163            let export = format!("export {key}={value}\n");
164            log::debug!("{export}");
165            tmc_params_file
166                .write_all(export.as_bytes())
167                .map_err(|e| FileError::FileWrite(tmc_params_path.clone(), e))?;
168        }
169    }
170
171    // make archive
172    log::debug!("creating submission archive");
173    let exercise_name = stub_clone_path.file_name();
174    let course_name = stub_clone_path.parent().and_then(Path::file_name);
175    let prefix = if no_archive_prefix {
176        PathBuf::new()
177    } else {
178        match (course_name, exercise_name) {
179            (Some(course_name), Some(exercise_name)) => Path::new(course_name).join(exercise_name),
180            _ => {
181                log::warn!(
182                    "was not able to find exercise and/or course name from clone path {}",
183                    stub_clone_path.display()
184                );
185                PathBuf::new()
186            }
187        }
188    };
189    let archive_file = file_util::create_file(target_path)?;
190    match output_format {
191        Compression::Tar => {
192            let mut archive = tar::Builder::new(archive_file);
193            for entry in WalkDir::new(&extract_dest_path)
194                .into_iter()
195                .filter_entry(|e| e.file_name() != file_util::LOCK_FILE_NAME)
196                .skip(1)
197            {
198                let entry = entry?;
199                let entry_path = entry.path();
200                let stripped = prefix.join(
201                    entry_path
202                        .strip_prefix(&extract_dest_path)
203                        .expect("the entry is inside dest"),
204                );
205                log::debug!(
206                    "adding {} to tar at {}",
207                    entry_path.display(),
208                    stripped.display()
209                );
210                if entry_path.is_dir() {
211                    archive
212                        .append_dir(&stripped, entry_path)
213                        .map_err(|e| LangsError::TarAppend(entry_path.to_path_buf(), e))?;
214                } else {
215                    archive
216                        .append_path_with_name(entry_path, stripped.to_string_lossy().as_ref())
217                        .map_err(|e| LangsError::TarAppend(entry_path.to_path_buf(), e))?;
218                }
219            }
220            archive
221                .finish()
222                .map_err(|e| LangsError::TarAppend(extract_dest_path.clone(), e))?;
223        }
224        Compression::Zip => {
225            let mut archive = ZipWriter::new(archive_file);
226            for entry in WalkDir::new(&extract_dest_path)
227                .into_iter()
228                .filter_entry(|e| e.file_name() != file_util::LOCK_FILE_NAME)
229                .skip(1)
230            {
231                let entry = entry?;
232                let entry_path = entry.path();
233                let stripped = prefix.join(
234                    entry_path
235                        .strip_prefix(&extract_dest_path)
236                        .expect("the entry is inside dest"),
237                );
238                log::debug!(
239                    "adding {} to zip at {}",
240                    entry_path.display(),
241                    stripped.display()
242                );
243                if entry_path.is_dir() {
244                    archive.add_directory(
245                        stripped.to_string_lossy(),
246                        SimpleFileOptions::default().unix_permissions(0o755),
247                    )?;
248                } else {
249                    archive.start_file(
250                        stripped.to_string_lossy(),
251                        SimpleFileOptions::default().unix_permissions(0o755),
252                    )?;
253                    let mut file = file_util::open_file(entry_path)?;
254                    std::io::copy(&mut file, &mut archive)
255                        .map_err(|e| LangsError::TarAppend(entry_path.to_path_buf(), e))?;
256                }
257            }
258            archive.finish()?;
259        }
260        Compression::TarZstd => {
261            let buf = Cursor::new(vec![]);
262            let mut archive = tar::Builder::new(buf);
263            for entry in WalkDir::new(&extract_dest_path)
264                .into_iter()
265                .filter_entry(|e| e.file_name() != file_util::LOCK_FILE_NAME)
266                .skip(1)
267            {
268                let entry = entry?;
269                let entry_path = entry.path();
270                let stripped = prefix.join(
271                    entry_path
272                        .strip_prefix(&extract_dest_path)
273                        .expect("the entry is inside dest"),
274                );
275                log::debug!(
276                    "adding {} to zip at {}",
277                    entry_path.display(),
278                    stripped.display()
279                );
280                if entry_path.is_dir() {
281                    archive
282                        .append_dir(stripped.to_string_lossy().as_ref(), entry_path)
283                        .map_err(|e| LangsError::TarAppend(entry_path.to_path_buf(), e))?;
284                } else {
285                    archive
286                        .append_path_with_name(entry_path, stripped.to_string_lossy().as_ref())
287                        .map_err(|e| LangsError::TarAppend(entry_path.to_path_buf(), e))?;
288                }
289            }
290
291            archive.finish().map_err(LangsError::TarFinish)?;
292            let mut tar = archive.into_inner().map_err(LangsError::TarIntoInner)?;
293            tar.set_position(0); // reset the cursor
294            zstd::stream::copy_encode(tar, archive_file, 0).map_err(LangsError::Zstd)?;
295        }
296    }
297
298    // get sandbox image
299    let sandbox_image = match TmcProjectYml::load(stub_clone_path)?.and_then(|c| c.sandbox_image) {
300        Some(sandbox_image) => sandbox_image,
301        None => crate::get_default_sandbox_image(stub_clone_path)?.to_string(),
302    };
303    Ok(sandbox_image)
304}

306fn extract_with_filter<F: Fn(&Path) -> bool>(
307    plugin: PluginType,
308    archive: &Path,
309    compression: Compression,
310    exclude_filter: F,
311    dest: &Path,
312    naive: bool,
313) -> Result<(), LangsError> {
314    let file = file_util::open_file(archive)?;
315    let mut zip = Archive::new(file, compression)?;
316    let project_dir_in_archive = if naive {
317        Ok(PathBuf::new())
318    } else {
319        plugin.safe_find_project_dir_in_archive(&mut zip)
320    }?;
321
322    let mut iter = zip.iter()?;
323    loop {
324        let next = iter.with_next::<(), _>(|mut file| {
325            if file.is_file() {
326                if let Ok(path) = file.path()?.strip_prefix(&project_dir_in_archive) {
327                    if exclude_filter(path) {
328                        // path component on ignore list
329                        return Ok(Continue(()));
330                    };
331                    let target = dest.join(path);
332                    file_util::read_to_file(&mut file, &target)?;
333                }
334            }
335            Ok(Continue(()))
336        });
337        match next? {
338            Continue(_) => continue,
339            Break(_) => break,
340        }
341    }
342    Ok(())
343}

373        let temp = tempfile::tempdir().unwrap();
374        let output_archive = temp.path().join("output.tar");

381        prepare_submission(
382            PrepareSubmission {
383                archive: Path::new(zip),
384                compression: Compression::Zip,
385                extract_naively: false,

393        )
394        .unwrap();
395        assert!(output_archive.exists());
396
397        let output_file = file_util::open_file(&output_archive).unwrap();
398        let mut archive = tar::Archive::new(output_file);
399        let output_extracted = temp.path().join("output");
400        archive.unpack(&output_extracted).unwrap();
401        for entry in WalkDir::new(temp.path()) {
402            log::debug!("file {}", entry.unwrap().path().display());
403        }
404        (temp, output_extracted)
405    }

678        file_util::write_to_file(b"key: stub", stub_ex_dir.join(".tmcproject.yml")).unwrap();
679

681        let sub_zip_path = temp.path().join("submission.zip");

688            .unwrap();

691            .unwrap();
692        std::io::Write::write_all(&mut zw, b"key: student\n").unwrap();
693        zw.finish().unwrap();

697            PrepareSubmission {

699                compression: Compression::Zip,
700                extract_naively: true,

702            &output_arch,
703            false,
704            TmcParams::new(),
705            &stub_ex_dir,
706            None,
707            Compression::Tar,
708        )
709        .unwrap();

718        // Verify .tmcproject.yml content is from stub, not student

734                .exists()
735        );
736        assert!(
737            output_extracted
738                .join("some_course/PythonExercise/test/test_greeter.py")
739                .exists()
740        );
741        assert!(
742            output_extracted
743                .join("some_course/PythonExercise/__init__.py")
744                .exists()
745        );

789        init();
790
791        let temp = tempfile::tempdir().unwrap();
792        let clone_root = temp.path().join("some_course");
793        file_util::create_dir_all(&clone_root).unwrap();
794
795        // Copy the Maven exercise to our temp directory
796        let src_clone = Path::new(MAVEN_CLONE);
797        file_util::copy(src_clone, &clone_root).unwrap();
798
799        let exercise_dir = clone_root.join("MavenExercise");
800
801        // Create a file in the root directory of the exercise (simulating repo file)
802        let repo_file_path = exercise_dir.join("foo.txt");
803        file_util::write_to_file(b"repohello", &repo_file_path).unwrap();
804
805        // Create a submission zip that also has foo.txt but with different content
806        // We need to create a proper Maven project structure that the plugin can understand
807        let sub_zip_path = temp.path().join("submission.zip");
808        let sub_zip_file = file_util::create_file(&sub_zip_path).unwrap();
809        let mut zw = zip::ZipWriter::new(sub_zip_file);
810        let opts = SimpleFileOptions::default();
811
812        // Create the Maven project structure
813        zw.add_directory("MavenExercise", opts).unwrap();
814        zw.add_directory("MavenExercise/src", opts).unwrap();
815        zw.add_directory("MavenExercise/src/main", opts).unwrap();
816        zw.add_directory("MavenExercise/src/main/java", opts)
817            .unwrap();
818
819        // Add the student's modified file
820        zw.start_file("MavenExercise/foo.txt", opts).unwrap();
821        std::io::Write::write_all(&mut zw, b"submissionhello").unwrap();
822
823        // Add a source file
824        zw.start_file("MavenExercise/src/main/java/SimpleStuff.java", opts)
825            .unwrap();
826        std::io::Write::write_all(&mut zw, b"public class SimpleStuff { }").unwrap();
827
828        zw.finish().unwrap();
829
830        let output_arch = temp.path().join("out.tar");
831        prepare_submission(
832            PrepareSubmission {
833                archive: &sub_zip_path,
834                compression: Compression::Zip,
835                extract_naively: false,
836            },
837            &output_arch,
838            false,
839            TmcParams::new(),
840            &exercise_dir,
841            None,
842            Compression::Tar,
843        )
844        .unwrap();
845        assert!(output_arch.exists());
846
847        // Unpack and verify that foo.txt content is from the repo, not submission
848        let output_file = file_util::open_file(&output_arch).unwrap();
849        let mut archive = tar::Archive::new(output_file);
850        let output_extracted = temp.path().join("output");
851        archive.unpack(&output_extracted).unwrap();
852
853        // Verify foo.txt exists and has content from repo, not submission
854        let foo_file = output_extracted.join("some_course/MavenExercise/foo.txt");
855        assert!(
856            foo_file.exists(),
857            "foo.txt should be included in the archive"
858        );
859        let content = fs::read_to_string(foo_file).unwrap();
860        assert_eq!(
861            content, "repohello",
862            "Should use repo content, not submission content"
863        );
864    }
