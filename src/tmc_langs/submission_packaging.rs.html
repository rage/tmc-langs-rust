25/// Prepares a submission for further processing.
26/// Returns the sandbox image to be used for the submission.
27pub fn prepare_submission(
28    submission: PrepareSubmission,
29    target_path: &Path,
30    no_archive_prefix: bool,
31    tmc_params: TmcParams,
32    stub_clone_path: &Path,
33    stub_archive: Option<(&Path, Compression)>,
34    output_format: Compression,
35) -> Result<String, LangsError> {
36    // FIXME: workaround for unknown issues when prepare_submission is ran multiple times in parallel
37    let _m = MUTEX.lock().map_err(|_| LangsError::MutexError)?;
38    log::debug!("preparing submission for {}", submission.archive.display());
39
40    let plugin = PluginType::from_exercise(stub_clone_path)?;
41
42    let extract_dest = tempfile::tempdir().map_err(LangsError::TempDir)?;
43    let extract_dest_path = extract_dest.path().to_path_buf();
44
45    // extract base
46    log::debug!("extracting stub");
47    let ignore_list = [
48        ".DS_Store",
49        "desktop.ini",
50        "Thumbs.db",
51        ".directory",
52        "__MACOSX",
53        file_util::LOCK_FILE_NAME,
54    ];
55    if let Some((stub_zip, compression)) = stub_archive {
56        // if defined, extract and use as the base
57        extract_with_filter(
58            plugin,
59            stub_zip,
60            compression,
61            |path| {
62                path.components().any(|c| {
63                    c.as_os_str()
64                        .to_str()
65                        .map(|s| ignore_list.contains(&s))
66                        .unwrap_or_default()
67                })
68            },
69            &extract_dest_path,
70            false,
71        )?;
72    } else {
73        // else, copy clone path
74        for entry in WalkDir::new(stub_clone_path).min_depth(1) {
75            let entry = entry?;
76
77            if entry.path().components().any(|c| {
78                c.as_os_str()
79                    .to_str()
80                    .map(|s| ignore_list.contains(&s))
81                    .unwrap_or_default()
82            }) {
83                // path component on ignore list
84                continue;
85            }
86
87            let relative_path = entry
88                .path()
89                .strip_prefix(stub_clone_path)
90                .expect("entry is in stub clone path");
91            let target_path = extract_dest_path.join(relative_path);
92            if entry.path().is_file() {
93                file_util::copy(entry.path(), target_path)?;
94            } else if entry.path().is_dir() {
95                file_util::create_dir(target_path)?;
96            }
97        }
98    }
99
100    // extract student files from submission over base
101    log::debug!("extracting student files");
102    let file = file_util::open_file(submission.archive)?;
103    if submission.extract_naively {
104        extract_project_overwrite(file, &extract_dest_path, submission.compression)?;
105    } else {
106        plugin.extract_student_files(file, submission.compression, &extract_dest_path)?;
107    }
108
109    // extract ide files
110    log::debug!("extracting ide files");
111    let ide_files = [
112        // netbeans
113        "nbproject",
114        // eclipse
115        ".classpath",
116        ".project",
117        ".settings",
118        // idea
119        ".idea",
120    ];
121    extract_with_filter(
122        plugin,
123        submission.archive,
124        submission.compression,
125        |path| {
126            path.components().all(|c| {
127                c.as_os_str()
128                    .to_str()
129                    .map(|s| !ide_files.contains(&s))
130                    .unwrap_or_default()
131            })
132        },
133        &extract_dest_path,
134        submission.extract_naively,
135    )?;
136
137    // write tmc params
138    if tmc_params.0.is_empty() {
139        log::debug!("no tmc params to write");
140    } else {
141        log::debug!("writing .tmcparams");
142        let tmc_params_path = extract_dest_path.join(".tmcparams");
143        let mut tmc_params_file = file_util::create_file(&tmc_params_path)?;
144        for (key, value) in tmc_params.0 {
145            // todo handle arrays, shell escapes
146            let export = format!("export {key}={value}\n");
147            log::debug!("{export}");
148            tmc_params_file
149                .write_all(export.as_bytes())
150                .map_err(|e| FileError::FileWrite(tmc_params_path.clone(), e))?;
151        }
152    }
153
154    // make archive
155    log::debug!("creating submission archive");
156    let exercise_name = stub_clone_path.file_name();
157    let course_name = stub_clone_path.parent().and_then(Path::file_name);
158    let prefix = if no_archive_prefix {
159        PathBuf::new()
160    } else {
161        match (course_name, exercise_name) {
162            (Some(course_name), Some(exercise_name)) => Path::new(course_name).join(exercise_name),
163            _ => {
164                log::warn!(
165                    "was not able to find exercise and/or course name from clone path {}",
166                    stub_clone_path.display()
167                );
168                PathBuf::new()
169            }
170        }
171    };
172    let archive_file = file_util::create_file(target_path)?;
173    match output_format {
174        Compression::Tar => {
175            let mut archive = tar::Builder::new(archive_file);
176            for entry in WalkDir::new(&extract_dest_path)
177                .into_iter()
178                .filter_entry(|e| e.file_name() != file_util::LOCK_FILE_NAME)
179                .skip(1)
180            {
181                let entry = entry?;
182                let entry_path = entry.path();
183                let stripped = prefix.join(
184                    entry_path
185                        .strip_prefix(&extract_dest_path)
186                        .expect("the entry is inside dest"),
187                );
188                log::debug!(
189                    "adding {} to tar at {}",
190                    entry_path.display(),
191                    stripped.display()
192                );
193                if entry_path.is_dir() {
194                    archive
195                        .append_dir(&stripped, entry_path)
196                        .map_err(|e| LangsError::TarAppend(entry_path.to_path_buf(), e))?;
197                } else {
198                    archive
199                        .append_path_with_name(entry_path, stripped.to_string_lossy().as_ref())
200                        .map_err(|e| LangsError::TarAppend(entry_path.to_path_buf(), e))?;
201                }
202            }
203            archive
204                .finish()
205                .map_err(|e| LangsError::TarAppend(extract_dest_path.clone(), e))?;
206        }
207        Compression::Zip => {
208            let mut archive = ZipWriter::new(archive_file);
209            for entry in WalkDir::new(&extract_dest_path)
210                .into_iter()
211                .filter_entry(|e| e.file_name() != file_util::LOCK_FILE_NAME)
212                .skip(1)
213            {
214                let entry = entry?;
215                let entry_path = entry.path();
216                let stripped = prefix.join(
217                    entry_path
218                        .strip_prefix(&extract_dest_path)
219                        .expect("the entry is inside dest"),
220                );
221                log::debug!(
222                    "adding {} to zip at {}",
223                    entry_path.display(),
224                    stripped.display()
225                );
226                if entry_path.is_dir() {
227                    archive.add_directory(
228                        stripped.to_string_lossy(),
229                        SimpleFileOptions::default().unix_permissions(0o755),
230                    )?;
231                } else {
232                    archive.start_file(
233                        stripped.to_string_lossy(),
234                        SimpleFileOptions::default().unix_permissions(0o755),
235                    )?;
236                    let mut file = file_util::open_file(entry_path)?;
237                    std::io::copy(&mut file, &mut archive)
238                        .map_err(|e| LangsError::TarAppend(entry_path.to_path_buf(), e))?;
239                }
240            }
241            archive.finish()?;
242        }
243        Compression::TarZstd => {
244            let buf = Cursor::new(vec![]);
245            let mut archive = tar::Builder::new(buf);
246            for entry in WalkDir::new(&extract_dest_path)
247                .into_iter()
248                .filter_entry(|e| e.file_name() != file_util::LOCK_FILE_NAME)
249                .skip(1)
250            {
251                let entry = entry?;
252                let entry_path = entry.path();
253                let stripped = prefix.join(
254                    entry_path
255                        .strip_prefix(&extract_dest_path)
256                        .expect("the entry is inside dest"),
257                );
258                log::debug!(
259                    "adding {} to zip at {}",
260                    entry_path.display(),
261                    stripped.display()
262                );
263                if entry_path.is_dir() {
264                    archive
265                        .append_dir(stripped.to_string_lossy().as_ref(), entry_path)
266                        .map_err(|e| LangsError::TarAppend(entry_path.to_path_buf(), e))?;
267                } else {
268                    archive
269                        .append_path_with_name(entry_path, stripped.to_string_lossy().as_ref())
270                        .map_err(|e| LangsError::TarAppend(entry_path.to_path_buf(), e))?;
271                }
272            }
273
274            archive.finish().map_err(LangsError::TarFinish)?;
275            let mut tar = archive.into_inner().map_err(LangsError::TarIntoInner)?;
276            tar.set_position(0); // reset the cursor
277            zstd::stream::copy_encode(tar, archive_file, 0).map_err(LangsError::Zstd)?;
278        }
279    }
280
281    // get sandbox image
282    let sandbox_image = match TmcProjectYml::load(stub_clone_path)?.and_then(|c| c.sandbox_image) {
283        Some(sandbox_image) => sandbox_image,
284        None => crate::get_default_sandbox_image(stub_clone_path)?.to_string(),
285    };
286    Ok(sandbox_image)
287}

289fn extract_with_filter<F: Fn(&Path) -> bool>(
290    plugin: PluginType,
291    archive: &Path,
292    compression: Compression,
293    exclude_filter: F,
294    dest: &Path,
295    naive: bool,
296) -> Result<(), LangsError> {
297    let file = file_util::open_file(archive)?;
298    let mut zip = Archive::new(file, compression)?;
299    let project_dir_in_archive = if naive {
300        PathBuf::new()
301    } else {
302        plugin.find_project_dir_in_archive(&mut zip)?
303    };
304
305    let mut iter = zip.iter()?;
306    loop {
307        let next = iter.with_next::<(), _>(|mut file| {
308            if file.is_file() {
309                if let Ok(path) = file.path()?.strip_prefix(&project_dir_in_archive) {
310                    if exclude_filter(path) {
311                        // path component on ignore list
312                        return Ok(Continue(()));
313                    };
314                    let target = dest.join(path);
315                    file_util::read_to_file(&mut file, &target)?;
316                }
317            }
318            Ok(Continue(()))
319        });
320        match next? {
321            Continue(_) => continue,
322            Break(_) => break,
323        }
324    }
325    Ok(())
326}

355    fn generic_submission(clone: &str, zip: &str) -> (TempDir, PathBuf) {
356        let temp = tempfile::tempdir().unwrap();
357        let output_archive = temp.path().join("output.tar");
358
359        let mut tmc_params = TmcParams::new();
360        tmc_params.insert_string("param_one", "value_one").unwrap();
361        tmc_params
362            .insert_array("param_two", vec!["value_two", "value_three"])
363            .unwrap();
364        prepare_submission(
365            PrepareSubmission {
366                archive: Path::new(zip),
367                compression: Compression::Zip,
368                extract_naively: false,
369            },
370            &output_archive,
371            false,
372            tmc_params,
373            Path::new(clone),
374            None,
375            Compression::Tar,
376        )
377        .unwrap();
378        assert!(output_archive.exists());
379
380        let output_file = file_util::open_file(&output_archive).unwrap();
381        let mut archive = tar::Archive::new(output_file);
382        let output_extracted = temp.path().join("output");
383        archive.unpack(&output_extracted).unwrap();
384        for entry in WalkDir::new(temp.path()) {
385            log::debug!("file {}", entry.unwrap().path().display());
386        }
387        (temp, output_extracted)
388    }
