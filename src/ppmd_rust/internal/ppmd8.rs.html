114impl<RC> PPMd8<RC> {
115    fn construct(
116        rc: RC,
117        max_order: u32,
118        mem_size: u32,
119        restore_method: RestoreMethod,
120    ) -> crate::Result<Self> {
121        let mut units2index = [0u8; 128];
122        let mut index2units = [0u8; 40];
123
124        let mut k = 0;
125        for i in 0..PPMD_NUM_INDEXES {
126            let mut step = if i >= 12 { 4 } else { (i >> 2) + 1 };
127            loop {
128                units2index[k as usize] = i as u8;
129                k += 1;
130
131                step -= 1;
132                if step == 0 {
133                    break;
134                }
135            }
136            index2units[i as usize] = k as u8;
137        }
138
139        let mut ns2bs_index = [0u8; 256];
140        ns2bs_index[0] = (0 << 1) as u8;
141        ns2bs_index[1] = (1 << 1) as u8;
142        ns2bs_index[2..11].fill((2 << 1) as u8);
143        ns2bs_index[11..256].fill((3 << 1) as u8);
144
145        let mut ns2index = [0u8; 260];
146        for i in 0..5 {
147            ns2index[i as usize] = i as u8;
148        }
149
150        let mut m = 5;
151        let mut k = 1;
152        for i in 5..260 {
153            ns2index[i as usize] = m as u8;
154            k -= 1;
155            if k == 0 {
156                m += 1;
157                k = m - 4;
158            }
159        }
160
161        let align_offset = (4u32.wrapping_sub(mem_size)) & 3;
162        let total_size = (align_offset + mem_size) as usize;
163
164        let memory_layout = Layout::from_size_align(total_size, align_of::<usize>())
165            .expect("Failed to create memory layout");
166
167        let allocation = unsafe {
168            assert_ne!(total_size, 0);
169            NonNull::new(alloc_zeroed(memory_layout))
170        };
171
172        let Some(memory_ptr) = allocation else {
173            return Err(Error::MemoryAllocation);
174        };
175
176        let mut ppmd = Self {
177            min_context: NonNull::dangling(),
178            max_context: NonNull::dangling(),
179            found_state: NonNull::dangling(),
180            order_fall: 0,
181            init_esc: 0,
182            prev_success: 0,
183            max_order,
184            restore_method,
185            run_length: 0,
186            init_rl: 0,
187            size: mem_size,
188            glue_count: 0,
189            align_offset,
190            lo_unit: NonNull::dangling(),
191            hi_unit: NonNull::dangling(),
192            text: NonNull::dangling(),
193            units_start: NonNull::dangling(),
194            index2units,
195            units2index,
196            free_list: [TaggedOffset::null(); 38],
197            stamps: [0; 38],
198            ns2bs_index,
199            ns2index,
200            exp_escape: K_EXP_ESCAPE,
201            dummy_see: See::default(),
202            see: [[See::default(); 32]; 24],
203            bin_summ: [[0; 64]; 25],
204            memory_ptr,
205            memory_layout,
206            rc,
207        };
208
209        unsafe { ppmd.restart_model() };
210
211        Ok(ppmd)
212    }

218    unsafe fn insert_node(&mut self, mut node: NonNull<Node>, index: u32) {
219        unsafe {
220            node.as_mut().stamp = 0xFFFFFFFF;
221            node.as_mut().next = self.free_list[index as usize];
222            node.as_mut().nu = self.index2units[index as usize] as u32;
223            self.free_list[index as usize] = self.offset_for_ptr(node);
224            self.stamps[index as usize] = self.stamps[index as usize].wrapping_add(1);
225        }
226    }

228    unsafe fn remove_node(&mut self, index: u32) -> NonNull<Node> {
229        let index = index as usize;
230        let node_offset = self.free_list[index];
231        let node = node_offset.as_ptr::<Node, _>(self);
232        self.free_list[index] = node.as_ref().next;
233        self.stamps[index] = self.stamps[index].wrapping_sub(1);
234        node
235    }

237    unsafe fn split_block(&mut self, mut ptr: NonNull<u8>, old_index: u32, new_index: u32) {
238        unsafe {
239            let nu = self.index2units[old_index as usize] as u32
240                - self.index2units[new_index as usize] as u32;
241            ptr = ptr.offset(self.index2units[new_index as usize] as isize * UNIT_SIZE);
242            let mut index = self.units2index[(nu as usize) - 1] as u32;
243            if self.index2units[index as usize] as u32 != nu {
244                index -= 1;
245                let k = self.index2units[index as usize] as u32;
246                self.insert_node(
247                    ptr.offset((k * UNIT_SIZE as u32) as isize).cast(),
248                    nu.wrapping_sub(k).wrapping_sub(1),
249                );
250            }
251            self.insert_node(ptr.cast(), index);
252        }
253    }

255    /// We use the first u16 field of the 12-bytes as record type stamp.
256    /// State   { symbol: u8, freq: u8, .. : freq != 0
257    /// Context { num_stats: u16, ..       : num_stats != 0
258    /// Node    { stamp: u16               : stamp == 0 for free record
259    ///                                    : stamp == 1 for head record and guard
260    /// Last 12-bytes record in array is always containing the 12-bytes order-0 Context
261    /// record.
262    unsafe fn glue_free_blocks(&mut self) {
263        unsafe {
264            let mut n = TaggedOffset::null();
265
266            self.glue_count = (1 << 13) as u32;
267            self.stamps = [0; 38];
268
269            // We set guard NODE at lo_unit.
270            if self.lo_unit != self.hi_unit {
271                self.lo_unit.cast::<Node>().as_mut().stamp = 0;
272            }
273
274            self.glue_blocks(&mut n);
275            self.fill_list(n);
276        }
277    }

279    /// Glue free blocks.
280    unsafe fn glue_blocks(&mut self, n: *mut TaggedOffset) {
281        unsafe {
282            let mut prev = n;
283            for i in 0..PPMD_NUM_INDEXES {
284                let mut next = self.free_list[i as usize];
285                self.free_list[i as usize] = TaggedOffset::null();
286                while next.is_not_null() {
287                    let mut node = next.as_ptr::<Node, _>(self);
288                    let mut nu = node.as_ref().nu;
289                    *prev = next;
290                    next = node.as_ref().next;
291                    if nu != 0 {
292                        prev = &raw mut node.as_mut().next;
293                        loop {
294                            let node2 = node.offset(nu as isize).as_mut();
295                            if node2.stamp != EMPTY_NODE {
296                                break;
297                            }
298                            nu += node2.nu;
299                            (*node.as_ptr()).nu = nu;
300                            node2.nu = 0;
301                        }
302                    }
303                }
304            }
305            *prev = TaggedOffset::null();
306        }
307    }

309    /// Fill lists of free blocks.
310    unsafe fn fill_list(&mut self, mut n: TaggedOffset) {
311        unsafe {
312            while n.is_not_null() {
313                let mut node = n.as_ptr::<Node, _>(self);
314                let mut nu = node.as_ref().nu;
315                n = node.as_ref().next;
316                if nu == 0 {
317                    continue;
318                }
319                while nu > 128 {
320                    self.insert_node(node.cast(), PPMD_NUM_INDEXES - 1);
321                    nu -= 128;
322                    node = node.offset(128);
323                }
324                let mut index = self.units2index[(nu as usize) - 1] as u32;
325                if self.index2units[index as usize] as u32 != nu {
326                    index -= 1;
327                    let k = self.index2units[index as usize] as u32;
328                    self.insert_node(
329                        node.offset(k as isize).cast(),
330                        nu.wrapping_sub(k).wrapping_sub(1),
331                    );
332                }
333                self.insert_node(node.cast(), index);
334            }
335        }
336    }

338    #[inline(never)]
339    unsafe fn alloc_units_rare(&mut self, index: u32) -> Option<NonNull<u8>> {
340        unsafe {
341            if self.glue_count == 0 {
342                self.glue_free_blocks();
343                if self.free_list[index as usize].is_not_null() {
344                    return Some(self.remove_node(index).cast());
345                }
346            }
347
348            let mut i = index;
349
350            loop {
351                i += 1;
352                if i == PPMD_NUM_INDEXES {
353                    let num_bytes = self.index2units[index as usize] as u32 * UNIT_SIZE as u32;
354                    let us = self.units_start;
355                    self.glue_count -= 1;
356                    return if us.offset_from(self.text) as u32 > num_bytes {
357                        self.units_start = us.offset(-(num_bytes as isize));
358                        Some(self.units_start)
359                    } else {
360                        None
361                    };
362                }
363                if self.free_list[i as usize].is_not_null() {
364                    break;
365                }
366            }
367
368            let block = self.remove_node(i).cast();
369            self.split_block(block, i, index);
370            Some(block)
371        }
372    }

374    unsafe fn alloc_units(&mut self, index: u32) -> Option<NonNull<u8>> {
375        unsafe {
376            if self.free_list[index as usize].is_not_null() {
377                return Some(self.remove_node(index).cast());
378            }
379            let num_bytes = self.index2units[index as usize] as u32 * UNIT_SIZE as u32;
380            let lo = self.lo_unit;
381            if self.hi_unit.offset_from(lo) as u32 >= num_bytes {
382                self.lo_unit = lo.offset(num_bytes as isize);
383                return Some(lo);
384            }
385            self.alloc_units_rare(index)
386        }
387    }

389    unsafe fn shrink_units(
390        &mut self,
391        old_ptr: NonNull<State>,
392        old_nu: u32,
393        new_nu: u32,
394    ) -> NonNull<u8> {
395        unsafe {
396            let old_ptr = old_ptr.cast();
397
398            let i0 = self.units2index[(old_nu as usize) - 1] as u32;
399            let i1 = self.units2index[(new_nu as usize) - 1] as u32;
400
401            if i0 == i1 {
402                return old_ptr;
403            }
404
405            if self.free_list[i1 as usize].is_not_null() {
406                let ptr = self.remove_node(i1).cast();
407                std::ptr::copy(
408                    old_ptr.as_ptr(),
409                    ptr.as_ptr(),
410                    new_nu as usize * UNIT_SIZE as usize,
411                );
412                self.insert_node(old_ptr.cast(), i0);
413                return ptr;
414            }
415
416            self.split_block(old_ptr, i0, i1);
417
418            old_ptr
419        }
420    }

438    unsafe fn expand_text_area(&mut self) {
439        unsafe {
440            let mut count: [u32; 38] = [0; 38];
441
442            if self.lo_unit != self.hi_unit {
443                self.lo_unit.cast::<Node>().as_mut().stamp = 0;
444            }
445
446            let mut node = self.units_start.cast::<Node>();
447            while node.as_ref().stamp == EMPTY_NODE {
448                let nu = node.as_ref().nu;
449                node.as_mut().stamp = 0;
450                count[self.units2index[(nu as usize) - 1] as usize] += 1;
451                node = node.offset(nu as isize);
452            }
453            self.units_start = node.cast();
454
455            for i in 0..PPMD_NUM_INDEXES {
456                let mut cnt = count[i as usize];
457                if cnt != 0 {
458                    let mut prev = &raw mut self.free_list[i as usize];
459                    let mut n = *prev;
460                    self.stamps[i as usize] = self.stamps[i as usize].wrapping_sub(cnt);
461                    loop {
462                        let mut node = n.as_ptr::<Node, _>(self);
463                        n = node.as_ref().next;
464                        if node.as_ref().stamp != 0 {
465                            prev = &raw mut node.as_mut().next;
466                            continue;
467                        }
468
469                        *prev = n;
470                        cnt -= 1;
471                        if cnt == 0 {
472                            break;
473                        }
474                    }
475                }
476            }
477        }
478    }

480    #[inline(never)]
481    unsafe fn restart_model(&mut self) {
482        unsafe {
483            self.free_list = [TaggedOffset::null(); 38];
484            self.stamps = [0; 38];
485
486            self.text = self.memory_ptr.offset(self.align_offset as isize);
487            self.hi_unit = self.text.offset(self.size as isize);
488            self.units_start = self
489                .hi_unit
490                .offset(-(self.size as isize / 8 / UNIT_SIZE * 7 * UNIT_SIZE));
491            self.lo_unit = self.units_start;
492            self.glue_count = 0;
493
494            self.order_fall = self.max_order;
495            self.init_rl = -(if self.max_order < 12 {
496                self.max_order as i32
497            } else {
498                12
499            }) - 1;
500            self.run_length = self.init_rl;
501            self.prev_success = 0;
502
503            self.hi_unit = self.hi_unit.offset(-UNIT_SIZE);
504            let mut mc = self.hi_unit.cast::<Context>();
505            let s = self.lo_unit.cast::<State>();
506
507            self.lo_unit = self.lo_unit.offset((256 / 2) * UNIT_SIZE);
508            self.min_context = mc;
509            self.max_context = self.min_context;
510            self.found_state = s;
511
512            {
513                let mc = mc.as_mut();
514                mc.flags = 0;
515                mc.num_stats = (256 - 1) as u8;
516                mc.union2.summ_freq = (256 + 1) as u16;
517                mc.union4.stats = self.offset_for_ptr(s);
518                mc.suffix = TaggedOffset::null();
519            }
520
521            (0..256).for_each(|i| {
522                let s = s.offset(i).as_mut();
523                s.symbol = i as u8;
524                s.freq = 1;
525                s.set_successor(TaggedOffset::null());
526            });
527
528            let mut i = 0;
529            (0..25).for_each(|m| {
530                while self.ns2index[i as usize] as usize == m {
531                    i += 1;
532                }
533
534                (0..8).for_each(|k| {
535                    let val = PPMD_BIN_SCALE - (K_INIT_BIN_ESC[k] as u32) / (i + 1);
536
537                    (0..64).step_by(8).for_each(|r| {
538                        self.bin_summ[m][k + r] = val as u16;
539                    });
540                });
541            });
542
543            let mut i = 0;
544            (0..24).for_each(|m| {
545                while self.ns2index[(i + 3) as usize] as usize == m + 3 {
546                    i += 1;
547                }
548
549                let summ = (2 * i + 5) << (PPMD_PERIOD_BITS - 4);
550
551                (0..32).for_each(|k| {
552                    let see = &mut self.see[m][k];
553                    see.summ = summ as u16;
554                    see.shift = (PPMD_PERIOD_BITS - 4) as u8;
555                    see.count = 7;
556                });
557            });
558
559            self.dummy_see.summ = 0; // unused
560            self.dummy_see.shift = PPMD_PERIOD_BITS as u8;
561            self.dummy_see.count = 64; // unused
562        }
563    }

565    /// This function is called when we remove some symbols (successors) in context.
566    /// It increases escape_freq for sum of all removed symbols.
567    unsafe fn refresh(&mut self, mut ctx: NonNull<Context>, old_nu: u32, mut scale: u32) {
568        unsafe {
569            let num_stats = ctx.as_ref().num_stats as u32;
570
571            let states = self.get_multi_state_stats(ctx);
572            let mut s = self
573                .shrink_units(states.cast(), old_nu, (num_stats + 2) >> 1)
574                .cast::<State>();
575            ctx.as_mut().union4.stats = self.offset_for_ptr(s);
576
577            scale |= (ctx.as_ref().union2.summ_freq as u32 >= 1 << 15) as u32;
578
579            let mut flags = Self::hi_bits_prepare(s.as_ref().symbol as u32);
580            let mut freq = s.as_ref().freq as u32;
581            let mut esc_freq = (ctx.as_ref().union2.summ_freq as u32) - freq;
582            freq = freq.wrapping_add(scale) >> scale;
583            let mut sum_freq = freq;
584            s.as_mut().freq = freq as u8;
585
586            let mut i = num_stats;
587
588            loop {
589                s = s.offset(1);
590                let mut freq = s.as_ref().freq as u32;
591                esc_freq -= freq;
592                freq += scale >> scale;
593                sum_freq += freq;
594                s.as_mut().freq = freq as u8;
595                flags |= Self::hi_bits_prepare(s.as_ref().symbol as u32);
596
597                i -= 1;
598                if i == 0 {
599                    break;
600                }
601            }
602
603            ctx.as_mut().union2.summ_freq = (sum_freq + ((esc_freq + scale) >> scale)) as u16;
604            ctx.as_mut().flags = (ctx.as_ref().flags as u32
605                & ((FLAG_PREV_HIGH as u32 + FLAG_RESCALED as u32 * scale)
606                    + Self::hi_bits_convert_3(flags))) as u8;
607        }
608    }

610    /// Reduces contexts.
611    ///
612    /// It converts successors at max_order to another contexts to NULL-successors.
613    /// It removes RAW-successors and NULL-successors that are not Order-0, and it
614    /// removes contexts when it has no successors. If the (multi_state.stats) is
615    /// close to (units_start), it moves it up.
616    unsafe fn cut_off(&mut self, mut ctx: NonNull<Context>, order: u32) -> TaggedOffset {
617        unsafe {
618            let mut ns = ctx.as_ref().num_stats as i32;
619
620            if ns == 0 {
621                let mut successor = ctx.as_ref().union4.state4.get_successor();
622                if successor.is_real_context(self) {
623                    if order < self.max_order {
624                        let context = self.get_context(successor);
625                        successor = self.cut_off(context, order + 1);
626                    } else {
627                        successor = TaggedOffset::null();
628                    }
629                    ctx.as_mut().union4.state4.set_successor(successor);
630                    if successor.is_not_null() || order <= 9 {
631                        // O_BOUND
632                        return self.offset_for_ptr(ctx);
633                    }
634                }
635                self.special_free_unit(ctx.cast());
636                return TaggedOffset::null();
637            }
638
639            let nu = (ns as u32).wrapping_add(2) >> 1;
640
641            let index = self.units2index[(nu as usize) - 1] as u32;
642            let stats_offset = ctx.as_ref().union4.stats;
643            let mut stats = stats_offset.as_ptr::<State, _>(self);
644
645            if stats.cast::<u8>().offset_from(self.units_start) <= (1 << 14)
646                && stats_offset.get_offset() <= self.free_list[index as usize].get_offset()
647            {
648                let ptr = self.remove_node(index);
649                ctx.as_mut().union4.stats = self.offset_for_ptr(ptr.cast::<State>());
650
651                std::ptr::copy(
652                    stats.cast().as_ptr(),
653                    ptr.as_ptr(),
654                    nu as usize * UNIT_SIZE as usize,
655                );
656
657                if stats.cast() != self.units_start {
658                    self.insert_node(stats.cast(), index);
659                } else {
660                    self.units_start = self.units_start.offset(
661                        (self.index2units[index as usize] as u32 * UNIT_SIZE as u32) as isize,
662                    );
663                }
664                stats = ptr.cast();
665            }
666
667            let mut s = stats.offset(ns as isize);
668
669            loop {
670                let successor = s.as_ref().get_successor();
671                if !successor.is_real_context(self) {
672                    let fresh = ns;
673                    ns -= 1;
674                    let mut s2 = stats.offset(fresh as isize);
675                    if order != 0 {
676                        if s != s2 {
677                            *s.as_mut() = *s2.as_ref();
678                        }
679                    } else {
680                        swap(s.as_mut(), s2.as_mut());
681                        s2.as_mut().set_successor(TaggedOffset::null());
682                    }
683                } else if order < self.max_order {
684                    let context = self.get_context(successor);
685                    s.as_mut().set_successor(self.cut_off(context, order + 1));
686                } else {
687                    s.as_mut().set_successor(TaggedOffset::null());
688                }
689
690                s = s.offset(-1);
691                if s < stats {
692                    break;
693                }
694            }
695
696            if ns != ctx.as_ref().num_stats as i32 && order != 0 {
697                if ns < 0 {
698                    self.free_units(stats.cast(), nu);
699                    self.special_free_unit(ctx.cast());
700                    return TaggedOffset::null();
701                }
702                ctx.as_mut().num_stats = ns as u8;
703                if ns == 0 {
704                    let sym = stats.as_ref().symbol;
705                    ctx.as_mut().flags = (((ctx.as_ref().flags & FLAG_PREV_HIGH) as u32)
706                        + Self::hi_bits_flag3(sym as u32))
707                        as u8;
708                    ctx.as_mut().union2.state2.symbol = sym;
709                    ctx.as_mut().union2.state2.freq =
710                        (((stats.as_ref().freq as u32) + 11) >> 3) as u8;
711                    s.as_mut().set_successor(stats.as_ref().get_successor());
712                    self.free_units(stats.cast(), nu);
713                } else {
714                    self.refresh(
715                        ctx,
716                        nu,
717                        (ctx.as_ref().union2.summ_freq as u32 > 16 * (ns as u32)) as u32,
718                    );
719                }
720            }
721
722            self.offset_for_ptr(ctx)
723        }
724    }

726    unsafe fn get_used_memory(&self) -> u32 {
727        unsafe {
728            let mut v = 0;
729
730            for i in 0..PPMD_NUM_INDEXES {
731                v *= self.stamps[i as usize] * self.index2units[i as usize] as u32;
732            }
733
734            self.size
735                - (self.hi_unit.offset_from(self.lo_unit) as u32)
736                - (self.units_start.offset_from(self.text) as u32)
737                - (v * UNIT_SIZE as u32)
738        }

741    unsafe fn restore_model(&mut self, ctx_error: NonNull<Context>) {
742        unsafe {
743            self.text = self.memory_ptr.offset(self.align_offset as isize);
744
745            // We go here in cases of error of allocation for context (c1)
746            // Order(min_context) < Order(ctx_error) <= Order(max_context)
747
748            let mut s;
749            let mut c = self.max_context;
750
751            // We remove last symbol from each of contexts [p.max_context ... ctx_error) contexts.
752            // So we roll back all created (symbols) before the error.
753            while c != ctx_error {
754                c.as_mut().num_stats -= 1;
755                if c.as_ref().num_stats as i32 == 0 {
756                    s = self.get_multi_state_stats(c);
757                    c.as_mut().flags = (((c.as_ref().flags & FLAG_PREV_HIGH) as u32)
758                        + Self::hi_bits_flag3(s.as_ref().symbol as u32))
759                        as u8;
760
761                    c.as_mut().union2.state2.symbol = s.as_ref().symbol;
762                    c.as_mut().union2.state2.symbol = (((s.as_ref().freq as u32) + 11) >> 3) as u8;
763                    c.as_mut()
764                        .union4
765                        .state4
766                        .set_successor(s.as_ref().get_successor());
767
768                    self.special_free_unit(s.cast());
769                } else {
770                    // refresh() can increase escape_freq on value of freq of last symbol,
771                    // that was added before error. So the largest possible increase for escape_freq
772                    // is (8) from value before model_update()
773                    self.refresh(c, ((c.as_ref().num_stats as u32) + 3) >> 1, 0);
774                }
775
776                c = self.get_context(c.as_ref().suffix);
777            }
778
779            // Increase escape_freq for context [ctx_error..p.MinContext)
780            while c != self.min_context {
781                if c.as_ref().num_stats as i32 == 0 {
782                    c.as_mut().union2.state2.freq =
783                        ((c.as_ref().union2.state2.freq as u32 + 1) >> 1) as u8;
784                } else {
785                    c.as_mut().union2.summ_freq = (c.as_ref().union2.summ_freq as i32 + 4) as u16;
786                    if c.as_ref().union2.summ_freq as i32 > 128 + 4 * c.as_ref().num_stats as i32 {
787                        self.refresh(c, (c.as_ref().num_stats as u32).wrapping_add(2) >> 1, 1);
788                    }
789                }
790
791                c = self.get_context(c.as_ref().suffix);
792            }
793
794            if self.restore_method == RestoreMethod::Restart
795                || self.get_used_memory() < self.size >> 1
796            {
797                self.restart_model();
798            } else {
799                while self.max_context.as_ref().suffix.is_not_null() {
800                    self.max_context = self.get_context(self.max_context.as_ref().suffix);
801                }
802                loop {
803                    self.cut_off(self.max_context, 0);
804                    self.expand_text_area();
805
806                    if self.get_used_memory() <= 3 * (self.size >> 2) {
807                        break;
808                    }
809                }
810                self.glue_count = 0;
811                self.order_fall = self.max_order;
812            }
813            self.min_context = self.max_context;
814        }
815    }

817    #[inline(never)]
818    unsafe fn create_successors(
819        &mut self,
820        skip: i32,
821        s1: &mut Option<NonNull<State>>,
822        mut c: NonNull<Context>,
823    ) -> Option<NonNull<Context>> {
824        unsafe {
825            let up_branch = self.found_state.as_ref().get_successor();
826            let mut num_ps = 0u32;
827            // Fixed over Shkarin's code. Maybe it could work without + 1 too.
828            let mut ps: [Option<NonNull<State>>; PPMD8_MAX_ORDER as usize + 1] =
829                [None; PPMD8_MAX_ORDER as usize + 1];
830
831            if skip == 0 {
832                let fresh = num_ps;
833                num_ps += 1;
834                ps[fresh as usize] = Some(self.found_state);
835            }
836
837            while c.as_ref().suffix.is_not_null() {
838                let mut s;
839                c = self.get_context(c.as_ref().suffix);
840
841                if let Some(state) = *s1 {
842                    s = state;
843                    *s1 = None;
844                } else if c.as_ref().num_stats as i32 != 0 {
845                    let sym = self.found_state.as_ref().symbol;
846                    s = self.get_multi_state_stats(c);
847
848                    while s.as_ref().symbol as i32 != sym as i32 {
849                        s = s.offset(1);
850                    }
851
852                    if (s.as_ref().freq) < MAX_FREQ - 9 {
853                        s.as_mut().freq += 1;
854                        c.as_mut().union2.summ_freq += 1;
855                    }
856                } else {
857                    s = self.get_single_state(c);
858                    s.as_mut().freq = (s.as_ref().freq as u32
859                        + ((self.get_context(c.as_ref().suffix).as_mut().num_stats == 0) as u32
860                            & ((s.as_ref().freq as u32) < 24) as u32))
861                        as u8;
862                }
863
864                let successor = s.as_ref().get_successor();
865                if successor != up_branch {
866                    c = self.get_context(successor);
867                    if num_ps == 0 {
868                        return Some(c);
869                    }
870                    break;
871                } else {
872                    let fresh = num_ps;
873                    num_ps = num_ps.wrapping_add(1);
874                    ps[fresh as usize] = Some(s);
875                }
876            }
877
878            let new_sym = *up_branch.as_ptr::<u8, _>(self).as_ref();
879            let new_offset = up_branch.get_offset() + 1;
880            let up_branch = TaggedOffset::from_bytes_offset(new_offset);
881
882            let flags = (Self::hi_bits_flag4(self.found_state.as_ref().symbol as u32)
883                + Self::hi_bits_flag3(new_sym as u32)) as u8;
884
885            let new_freq = if c.as_ref().num_stats as i32 == 0 {
886                c.as_ref().union2.state2.freq
887            } else {
888                let mut s = self.get_multi_state_stats(c);
889                while s.as_ref().symbol as i32 != new_sym as i32 {
890                    s = s.offset(1);
891                }
892                let cf = (s.as_ref().freq as u32) - 1;
893                let s0 = c.as_ref().union2.summ_freq as u32 - c.as_ref().num_stats as u32 - cf;
894                1 + (if 2 * cf <= s0 {
895                    (5 * cf > s0) as u32
896                } else {
897                    (cf + (2 * s0) - 3) / s0
898                }) as u8
899            };
900
901            loop {
902                let mut c1: NonNull<Context> = if self.hi_unit != self.lo_unit {
903                    self.hi_unit = self.hi_unit.offset(-UNIT_SIZE);
904                    self.hi_unit.cast()
905                } else if self.free_list[0].is_not_null() {
906                    self.remove_node(0).cast()
907                } else {
908                    self.alloc_units_rare(0)?.cast()
909                };
910
911                {
912                    let c1_mut = c1.as_mut();
913                    c1_mut.flags = flags;
914                    c1_mut.num_stats = 0;
915                    c1_mut.union2.state2.symbol = new_sym;
916                    c1_mut.union2.state2.freq = new_freq;
917                }
918
919                self.get_single_state(c1).as_mut().set_successor(up_branch);
920
921                c1.as_mut().suffix = self.offset_for_ptr(c);
922                num_ps = num_ps.wrapping_sub(1);
923
924                let mut state = ps[num_ps as usize].expect("successor not set");
925                state
926                    .as_mut()
927                    .set_successor(self.offset_for_ptr(c1.cast::<Context>()));
928
929                c = c1;
930                if num_ps == 0 {
931                    break;
932                }
933            }
934
935            Some(c)
936        }
937    }

939    unsafe fn reduce_order(
940        &mut self,
941        mut s1: Option<NonNull<State>>,
942        mut c: NonNull<Context>,
943    ) -> Option<NonNull<Context>> {
944        unsafe {
945            let mut s;
946            let c1 = c;
947            let up_branch = self.offset_for_ptr::<u8>(self.text.cast());
948
949            self.found_state.as_mut().set_successor(up_branch);
950            self.order_fall += 1;
951
952            loop {
953                if let Some(state) = s1 {
954                    c = self.get_context(c.as_ref().suffix);
955                    s = state;
956                    s1 = None;
957                } else {
958                    if c.as_ref().suffix.is_null() {
959                        return Some(c);
960                    }
961                    c = self.get_context(c.as_ref().suffix);
962                    if c.as_ref().num_stats != 0 {
963                        s = self.get_multi_state_stats(c);
964                        if s.as_ref().symbol as i32 != self.found_state.as_ref().symbol as i32 {
965                            loop {
966                                s = s.offset(1);
967
968                                if s.as_ref().symbol as i32
969                                    == self.found_state.as_ref().symbol as i32
970                                {
971                                    break;
972                                }
973                            }
974                        }
975                        if (s.as_ref().freq) < MAX_FREQ - 9 {
976                            s.as_mut().freq = (s.as_ref().freq as i32 + 2) as u8;
977                            c.as_mut().union2.summ_freq =
978                                (c.as_ref().union2.summ_freq as i32 + 2) as u16;
979                        }
980                    } else {
981                        s = self.get_single_state(c);
982                        s.as_mut().freq =
983                            (s.as_ref().freq as i32 + ((s.as_ref().freq as i32) < 32) as i32) as u8;
984                    }
985                }
986                if s.as_ref().get_successor().is_not_null() {
987                    break;
988                }
989                s.as_mut().set_successor(up_branch);
990                self.order_fall += 1;
991            }
992
993            if s.as_ref().get_successor().get_offset() <= up_branch.get_offset() {
994                let s2 = self.found_state;
995                self.found_state = s;
996                match self.create_successors(0, &mut None, c) {
997                    None => {
998                        s.as_mut().set_successor(TaggedOffset::null());
999                    }
1000                    Some(successor) => {
1001                        s.as_mut().set_successor(self.offset_for_ptr(successor));
1002                    }
1003                }
1004                self.found_state = s2;
1005            }
1006
1007            let successor = s.as_ref().get_successor();
1008            if self.order_fall == 1 && c1 == self.max_context {
1009                self.found_state.as_mut().set_successor(successor);
1010                self.text = self.text.offset(-1);
1011            }
1012            if successor.is_null() {
1013                return None;
1014            }
1015
1016            Some(self.get_context(successor))
1017        }
1018    }

1020    #[inline(never)]
1021    unsafe fn update_model(&mut self) {
1022        unsafe {
1023            let mut max_successor;
1024            let mut min_successor = self.found_state.as_ref().get_successor();
1025            let mut c;
1026            let f_freq = self.found_state.as_ref().freq as u32;
1027            let f_symbol = self.found_state.as_ref().symbol;
1028            let mut s: Option<NonNull<State>> = None;
1029
1030            if (self.found_state.as_ref().freq) < MAX_FREQ / 4
1031                && self.min_context.as_ref().suffix.is_not_null()
1032            {
1033                // Update frequencies in suffix Context
1034                c = self.get_context(self.min_context.as_ref().suffix);
1035                if c.as_ref().num_stats as i32 == 0 {
1036                    let mut state = self.get_single_state(c);
1037                    if (state.as_ref().freq as i32) < 32 {
1038                        state.as_mut().freq += 1;
1039                    }
1040                    s = Some(state);
1041                } else {
1042                    let sym = self.found_state.as_ref().symbol;
1043                    let mut state = self.get_multi_state_stats(c);
1044
1045                    if state.as_ref().symbol != sym {
1046                        while state.as_ref().symbol != sym {
1047                            state = state.offset(1);
1048                        }
1049                        if state.offset(0).as_ref().freq as i32
1050                            >= state.offset(-1).as_ref().freq as i32
1051                        {
1052                            swap(state.offset(0).as_mut(), state.offset(-1).as_mut());
1053                            state = state.offset(-1);
1054                        }
1055                    }
1056
1057                    if (state.as_ref().freq) < MAX_FREQ - 9 {
1058                        state.as_mut().freq = (state.as_ref().freq as i32 + 2) as u8;
1059                        c.as_mut().union2.summ_freq =
1060                            (c.as_ref().union2.summ_freq as i32 + 2) as u16;
1061                    }
1062
1063                    s = Some(state);
1064                }
1065            }
1066
1067            c = self.max_context;
1068            if self.order_fall == 0 && min_successor.is_not_null() {
1069                let Some(cs) = self.create_successors(1, &mut s, self.min_context) else {
1070                    self.found_state
1071                        .as_mut()
1072                        .set_successor(TaggedOffset::null());
1073                    self.restore_model(c);
1074                    return;
1075                };
1076
1077                self.found_state
1078                    .as_mut()
1079                    .set_successor(self.offset_for_ptr(cs));
1080                self.max_context = cs;
1081                self.min_context = self.max_context;
1082                return;
1083            }
1084
1085            let mut text = self.text;
1086            let mut fresh = text;
1087            text = text.offset(1);
1088            *fresh.as_mut() = self.found_state.as_ref().symbol;
1089            self.text = text;
1090            if text >= self.units_start {
1091                self.restore_model(c);
1092                return;
1093            }
1094            max_successor = self.offset_for_ptr::<u8>(text.cast());
1095
1096            if min_successor.is_null() {
1097                let Some(cs) = self.reduce_order(s, self.min_context) else {
1098                    self.restore_model(c);
1099                    return;
1100                };
1101                min_successor = self.offset_for_ptr(cs);
1102            } else if !min_successor.is_real_context(self) {
1103                let Some(cs) = self.create_successors(0, &mut s, self.min_context) else {
1104                    self.restore_model(c);
1105                    return;
1106                };
1107                min_successor = self.offset_for_ptr(cs);
1108            }
1109
1110            self.order_fall -= 1;
1111            if self.order_fall == 0 {
1112                max_successor = min_successor;
1113                self.text = self
1114                    .text
1115                    .offset(-((self.max_context != self.min_context) as isize));
1116            }
1117
1118            let flag = Self::hi_bits_flag3(f_symbol as u32) as u8;
1119            let ns = self.min_context.as_ref().num_stats as u32;
1120            let s0 = (self.min_context.as_ref().union2.summ_freq as u32) - ns - f_freq;
1121
1122            while c != self.min_context {
1123                let mut sum;
1124                let ns1 = c.as_ref().num_stats as u32;
1125                if ns1 != 0 {
1126                    if ns1 & 1 != 0 {
1127                        // Expand for one UNIT
1128                        let old_nu = (ns1 + 1) >> 1;
1129                        let i = self.units2index[(old_nu as usize) - 1] as u32;
1130                        if i != self.units2index[old_nu as usize] as u32 {
1131                            let Some(ptr) = self.alloc_units(i.wrapping_add(1)) else {
1132                                self.restore_model(c);
1133                                return;
1134                            };
1135                            let old_ptr = self.get_multi_state_stats(c);
1136                            std::ptr::copy(
1137                                old_ptr.cast().as_ptr(),
1138                                ptr.as_ptr(),
1139                                old_nu as usize * UNIT_SIZE as usize,
1140                            );
1141                            self.insert_node(old_ptr.cast(), i);
1142                            c.as_mut().union4.stats = self.offset_for_ptr(ptr.cast::<State>());
1143                        }
1144                    }
1145                    sum = c.as_ref().union2.summ_freq as u32;
1146                    // max increase of escape_freq is 1 here.
1147                    // An average increase is 1/3 per symbol
1148                    sum = sum.wrapping_add(((3 * ns1 + 1) < ns) as u32);
1149                } else {
1150                    let Some(s_ptr) = self.alloc_units(0) else {
1151                        self.restore_model(c);
1152                        return;
1153                    };
1154                    let mut s = s_ptr.cast::<State>();
1155
1156                    let mut freq = c.as_ref().union2.state2.freq as u32;
1157                    s.as_mut().symbol = c.as_ref().union2.state2.symbol;
1158                    s.as_mut()
1159                        .set_successor(c.as_ref().union4.state4.get_successor());
1160                    c.as_mut().union4.stats = self.offset_for_ptr(s);
1161
1162                    if freq < (MAX_FREQ as i32 / 4 - 1) as u32 {
1163                        freq <<= 1;
1164                    } else {
1165                        freq = (MAX_FREQ as i32 - 4) as u32;
1166                    }
1167                    s.as_mut().freq = freq as u8;
1168
1169                    sum = freq + self.init_esc + (ns > 2) as u32;
1170                }
1171
1172                let mut s = self.get_multi_state_stats(c).offset(ns1 as isize).offset(1);
1173                let mut cf = 2 * sum.wrapping_add(6) * f_freq;
1174                let sf = s0.wrapping_add(sum);
1175                s.as_mut().symbol = f_symbol;
1176                c.as_mut().num_stats = ns1.wrapping_add(1) as u8;
1177                s.as_mut().set_successor(max_successor);
1178                c.as_mut().flags = c.as_ref().flags | flag;
1179                if cf < 6 * sf {
1180                    cf = 1 + (cf > sf) as u32 + (cf >= 4 * sf) as u32;
1181                    sum = sum.wrapping_add(4);
1182                    // It can add (1, 2, 3) to escape_freq
1183                } else {
1184                    cf =
1185                        4u32 + (cf > 9 * sf) as u32 + (cf > 12 * sf) as u32 + (cf > 15 * sf) as u32;
1186                    sum = sum.wrapping_add(cf);
1187                }
1188
1189                c.as_mut().union2.summ_freq = sum as u16;
1190                s.as_mut().freq = cf as u8;
1191                c = self.get_context(c.as_ref().suffix);
1192            }
1193
1194            self.min_context = self.get_context(min_successor);
1195            self.max_context = self.min_context;
1196        }
1197    }

1199    #[inline(never)]
1200    unsafe fn rescale(&mut self) {
1201        let stats = self.get_multi_state_stats(self.min_context);
1202        let mut s = self.found_state;
1203
1204        // Sort the list by freq
1205        if s != stats {
1206            let tmp = *s.as_ref();
1207            loop {
1208                *s.offset(0).as_mut() = *s.offset(-1).as_ref();
1209                s = s.offset(-1);
1210                if s == stats {
1211                    break;
1212                }
1213            }
1214            *s.as_mut() = tmp;
1215        }
1216
1217        let mut sum_freq = s.as_ref().freq as u32;
1218        let mut esc_freq = (self.min_context.as_ref().union2.summ_freq as u32) - sum_freq;
1219
1220        let adder = (self.order_fall != 0) as u32;
1221
1222        sum_freq = (sum_freq + 4 + (adder)) >> 1;
1223        s.as_mut().freq = sum_freq as u8;
1224
1225        let mut i = self.min_context.as_ref().num_stats as u32;
1226
1227        loop {
1228            s = s.offset(1);
1229            let mut freq = s.as_ref().freq as u32;
1230            esc_freq -= freq;
1231            freq = freq.wrapping_add(adder) >> 1;
1232            sum_freq += freq;
1233            s.as_mut().freq = freq as u8;
1234
1235            if freq > s.offset(-1).as_ref().freq as u32 {
1236                let tmp = *s.as_ref();
1237                let mut s1 = s;
1238                loop {
1239                    *s1.offset(0).as_mut() = *s1.offset(-1).as_ref();
1240                    s1 = s1.offset(-1);
1241                    if !(s1 != stats && freq > s1.offset(-1).as_ref().freq as u32) {
1242                        break;
1243                    }
1244                }
1245                *s1.as_mut() = tmp;
1246            }
1247            i -= 1;
1248            if i == 0 {
1249                break;
1250            }
1251        }
1252
1253        if s.as_ref().freq == 0 {
1254            // Remove all items with freq == 0
1255            let mut i = 0;
1256            loop {
1257                i += 1;
1258                s = s.offset(-1);
1259                if s.as_ref().freq != 0 {
1260                    break;
1261                }
1262            }
1263
1264            esc_freq += i;
1265            let mut mc = self.min_context;
1266            let num_stats = mc.as_ref().num_stats as u32;
1267            let num_stats_new = num_stats - i;
1268            mc.as_mut().num_stats = num_stats_new as u8;
1269            let n0 = (num_stats + 2) >> 1;
1270
1271            if num_stats_new == 0 {
1272                let mut freq = (2u32 * (stats.as_ref().freq as u32)).div_ceil(esc_freq);
1273                if freq > (MAX_FREQ / 3) as u32 {
1274                    freq = (MAX_FREQ / 3) as u32;
1275                }
1276                mc.as_mut().flags = (((mc.as_ref().flags & FLAG_PREV_HIGH) as u32)
1277                    + Self::hi_bits_flag3(stats.as_ref().symbol as u32))
1278                    as u8;
1279
1280                s = self.get_single_state(mc);
1281                *s.as_mut() = *stats.as_ref();
1282                s.as_mut().freq = freq as u8;
1283                self.found_state = s;
1284                self.insert_node(stats.cast(), self.units2index[(n0 as usize) - 1] as u32);
1285                return;
1286            }
1287
1288            let n1 = (num_stats_new + 2) >> 1;
1289            if n0 != n1 {
1290                let shrunk = self.shrink_units(stats.cast(), n0, n1);
1291                mc.as_mut().union4.stats = self.offset_for_ptr(shrunk.cast::<State>());
1292            }
1293        }
1294
1295        let mc = self.min_context.as_mut();
1296        mc.union2.summ_freq = sum_freq.wrapping_add(esc_freq).wrapping_sub(esc_freq >> 1) as u16;
1297        mc.flags |= FLAG_RESCALED;
1298        self.found_state = mc.union4.stats.as_ptr(self);
1299    }

1301    unsafe fn make_esc_freq(&mut self, num_masked: u32, esc_freq: &mut u32) -> SeeSource {
1302        unsafe {
1303            let num_stats = self.min_context.as_ref().num_stats as u32;
1304
1305            if num_stats != 0xFF {
1306                let (base_context_idx, see_table_hash) =
1307                    self.calculate_see_table_hash(num_masked, num_stats);
1308
1309                let see = &mut self.see[base_context_idx][see_table_hash];
1310
1311                // If (see.summ) field is larger than 16-bit, we need only low 16 bits of summ.
1312                let summ = see.summ as u32;
1313                let r = summ >> see.shift as i32;
1314                see.summ = (summ - r) as u16;
1315                *esc_freq = r + (r == 0) as u32;
1316
1317                SeeSource::Table(base_context_idx, see_table_hash)
1318            } else {
1319                *esc_freq = 1;
1320                SeeSource::Dummy
1321            }
1322        }
1323    }

1325    unsafe fn calculate_see_table_hash(
1326        &mut self,
1327        num_masked: u32,
1328        num_stats: u32,
1329    ) -> (usize, usize) {
1330        unsafe {
1331            let mc = self.min_context;
1332            let base_context_idx = self.ns2index[(num_stats + 2) as usize] as usize - 3;
1333
1334            let suffix_context = self.get_context(mc.as_ref().suffix);
1335            let suffix_num_stats = suffix_context.as_ref().num_stats as u32;
1336            let summ_freq = mc.as_ref().union2.summ_freq as u32;
1337
1338            let freq_distribution_hash = (summ_freq > 11 * (num_stats + 1)) as usize;
1339            let context_hierarchy_hash =
1340                2 * ((2 * num_stats) < (suffix_num_stats + num_masked)) as usize;
1341            let symbol_characteristics_hash = mc.as_ref().flags as usize;
1342
1343            let see_table_hash =
1344                freq_distribution_hash + context_hierarchy_hash + symbol_characteristics_hash;
1345
1346            (base_context_idx, see_table_hash)
1347        }
1348    }

1358        unsafe {
1359            let successor = self.found_state.as_ref().get_successor();
1360            if self.order_fall == 0 && successor.is_real_context(self) {
1361                let context = self.get_context(successor);
1362                self.min_context = context;
1363                self.max_context = self.min_context;
1364            } else {

1370    unsafe fn update1(&mut self) {
1371        unsafe {
1372            let mut s = self.found_state;
1373            let mut freq = s.as_ref().freq as u32;
1374            freq += 4;
1375            self.min_context.as_mut().union2.summ_freq += 4;
1376            s.as_mut().freq = freq as u8;
1377            if freq > s.offset(-1).as_ref().freq as u32 {
1378                swap(s.as_mut(), s.offset(-1).as_mut());
1379                s = s.offset(-1);
1380                self.found_state = s;
1381                if freq > MAX_FREQ as u32 {
1382                    self.rescale();
1383                }
1384            }
1385            self.next_context();
1386        }
1387    }

1389    unsafe fn update1_0(&mut self) {
1390        unsafe {
1391            let mut s = self.found_state;
1392            let mut mc = self.min_context;
1393            let mut freq = s.as_ref().freq as u32;
1394            let summ_freq = mc.as_ref().union2.summ_freq as u32;
1395            self.prev_success = (2 * freq >= summ_freq) as u32; // Ppmd8 (>=)
1396            self.run_length += self.prev_success as i32;
1397            mc.as_mut().union2.summ_freq = (summ_freq + 4) as u16;
1398            freq += 4;
1399            s.as_mut().freq = freq as u8;
1400            if freq > MAX_FREQ as u32 {
1401                self.rescale();
1402            }
1403            self.next_context();
1404        }

1423        unsafe {
1424            let freq = s.as_ref().freq as u32;

1435    unsafe fn mask_symbols(char_mask: &mut [u8; 256], s: NonNull<State>, mut s2: NonNull<State>) {
1436        unsafe {
1437            char_mask[s.as_ref().symbol as usize] = 0;
1438            loop {
1439                let sym0 = s2.offset(0).as_ref().symbol as u32;
1440                let sym1 = s2.offset(1).as_ref().symbol as u32;
1441                s2 = s2.offset(2);
1442                char_mask[sym0 as usize] = 0;
1443                char_mask[sym1 as usize] = 0;
1444
1445                if s2 >= s {
1446                    break;
1447                }
1448            }
1449        }
