14#[cfg(target_arch = "x86_64")]
15const fn reg(input: [u32; 4]) -> __m128i {
16    // SAFETY: any valid [u32; 4] represents a valid __m128i
17    unsafe { core::mem::transmute(input) }
18}

20/// # Safety
21///
22/// The methods of this struct should only be used on x86_64 systems that support the `pclmulqdq`
23/// instruction, and SSE2 and SSE4.1 instruction sets.

25#[cfg(target_arch = "x86_64")]

51    #[target_feature(enable = "pclmulqdq", enable = "sse2", enable = "sse4.1")]
52    pub unsafe fn finish(self) -> u32 {
53        const CRC_MASK1: __m128i =
54            reg([0xFFFFFFFFu32, 0xFFFFFFFFu32, 0x00000000u32, 0x00000000u32]);
55
56        const CRC_MASK2: __m128i =
57            reg([0x00000000u32, 0xFFFFFFFFu32, 0xFFFFFFFFu32, 0xFFFFFFFFu32]);
58
59        const RK1_RK2: __m128i = reg([
60            0xccaa009e, 0x00000000, /* rk1 */
61            0x751997d0, 0x00000001, /* rk2 */
62        ]);
63
64        const RK5_RK6: __m128i = reg([
65            0xccaa009e, 0x00000000, /* rk5 */
66            0x63cd6124, 0x00000001, /* rk6 */
67        ]);
68
69        const RK7_RK8: __m128i = reg([
70            0xf7011640, 0x00000001, /* rk7 */
71            0xdb710640, 0x00000001, /* rk8 */
72        ]);
73
74        let [mut xmm_crc0, mut xmm_crc1, mut xmm_crc2, mut xmm_crc3] = self.fold;
75
76        /*
77         * k1
78         */
79        let mut crc_fold = RK1_RK2;
80
81        #[allow(unused_unsafe)] // because target features 1.1
82        unsafe {
83            let x_tmp0 = _mm_clmulepi64_si128(xmm_crc0, crc_fold, 0x10);
84            xmm_crc0 = _mm_clmulepi64_si128(xmm_crc0, crc_fold, 0x01);
85            xmm_crc1 = _mm_xor_si128(xmm_crc1, x_tmp0);
86            xmm_crc1 = _mm_xor_si128(xmm_crc1, xmm_crc0);
87
88            let x_tmp1 = _mm_clmulepi64_si128(xmm_crc1, crc_fold, 0x10);
89            xmm_crc1 = _mm_clmulepi64_si128(xmm_crc1, crc_fold, 0x01);
90            xmm_crc2 = _mm_xor_si128(xmm_crc2, x_tmp1);
91            xmm_crc2 = _mm_xor_si128(xmm_crc2, xmm_crc1);
92
93            let x_tmp2 = _mm_clmulepi64_si128(xmm_crc2, crc_fold, 0x10);
94            xmm_crc2 = _mm_clmulepi64_si128(xmm_crc2, crc_fold, 0x01);
95            xmm_crc3 = _mm_xor_si128(xmm_crc3, x_tmp2);
96            xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_crc2);
97
98            /*
99             * k5
100             */
101            crc_fold = RK5_RK6;
102
103            xmm_crc0 = xmm_crc3;
104            xmm_crc3 = _mm_clmulepi64_si128(xmm_crc3, crc_fold, 0);
105            xmm_crc0 = _mm_srli_si128(xmm_crc0, 8);
106            xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_crc0);
107
108            xmm_crc0 = xmm_crc3;
109            xmm_crc3 = _mm_slli_si128(xmm_crc3, 4);
110            xmm_crc3 = _mm_clmulepi64_si128(xmm_crc3, crc_fold, 0x10);
111            xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_crc0);
112            xmm_crc3 = _mm_and_si128(xmm_crc3, CRC_MASK2);
113
114            /*
115             * k7
116             */
117            xmm_crc1 = xmm_crc3;
118            xmm_crc2 = xmm_crc3;
119            crc_fold = RK7_RK8;
120
121            xmm_crc3 = _mm_clmulepi64_si128(xmm_crc3, crc_fold, 0);
122            xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_crc2);
123            xmm_crc3 = _mm_and_si128(xmm_crc3, CRC_MASK1);
124
125            xmm_crc2 = xmm_crc3;
126            xmm_crc3 = _mm_clmulepi64_si128(xmm_crc3, crc_fold, 0x10);
127            xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_crc2);
128            xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_crc1);
129
130            !(_mm_extract_epi32(xmm_crc3, 2) as u32)
131        }
132    }

134    unsafe fn fold_step<const N: usize>(&mut self) {
135        self.fold = core::array::from_fn(|i| match self.fold.get(i + N) {
136            Some(v) => *v,
137            None => unsafe { Self::step(self.fold[(i + N) - 4]) },
138        });
139    }
140
141    #[inline(always)]
142    unsafe fn step(input: __m128i) -> __m128i {
143        unsafe {
144            _mm_xor_si128(
145                _mm_clmulepi64_si128(input, Self::XMM_FOLD4, 0x01),
146                _mm_clmulepi64_si128(input, Self::XMM_FOLD4, 0x10),
147            )
148        }
149    }

151    unsafe fn partial_fold(&mut self, xmm_crc_part: __m128i, len: usize) {
152        const PSHUFB_SHF_TABLE: [__m128i; 15] = [
153            reg([0x84838281, 0x88878685, 0x8c8b8a89, 0x008f8e8d]), /* shl 15 (16 - 1)/shr1 */
154            reg([0x85848382, 0x89888786, 0x8d8c8b8a, 0x01008f8e]), /* shl 14 (16 - 3)/shr2 */
155            reg([0x86858483, 0x8a898887, 0x8e8d8c8b, 0x0201008f]), /* shl 13 (16 - 4)/shr3 */
156            reg([0x87868584, 0x8b8a8988, 0x8f8e8d8c, 0x03020100]), /* shl 12 (16 - 4)/shr4 */
157            reg([0x88878685, 0x8c8b8a89, 0x008f8e8d, 0x04030201]), /* shl 11 (16 - 5)/shr5 */
158            reg([0x89888786, 0x8d8c8b8a, 0x01008f8e, 0x05040302]), /* shl 10 (16 - 6)/shr6 */
159            reg([0x8a898887, 0x8e8d8c8b, 0x0201008f, 0x06050403]), /* shl  9 (16 - 7)/shr7 */
160            reg([0x8b8a8988, 0x8f8e8d8c, 0x03020100, 0x07060504]), /* shl  8 (16 - 8)/shr8 */
161            reg([0x8c8b8a89, 0x008f8e8d, 0x04030201, 0x08070605]), /* shl  7 (16 - 9)/shr9 */
162            reg([0x8d8c8b8a, 0x01008f8e, 0x05040302, 0x09080706]), /* shl  6 (16 -10)/shr10*/
163            reg([0x8e8d8c8b, 0x0201008f, 0x06050403, 0x0a090807]), /* shl  5 (16 -11)/shr11*/
164            reg([0x8f8e8d8c, 0x03020100, 0x07060504, 0x0b0a0908]), /* shl  4 (16 -12)/shr12*/
165            reg([0x008f8e8d, 0x04030201, 0x08070605, 0x0c0b0a09]), /* shl  3 (16 -13)/shr13*/
166            reg([0x01008f8e, 0x05040302, 0x09080706, 0x0d0c0b0a]), /* shl  2 (16 -14)/shr14*/
167            reg([0x0201008f, 0x06050403, 0x0a090807, 0x0e0d0c0b]), /* shl  1 (16 -15)/shr15*/
168        ];
169
170        unsafe {
171            let xmm_shl = PSHUFB_SHF_TABLE[len - 1];
172            let xmm_shr = _mm_xor_si128(xmm_shl, reg([0x80808080u32; 4]));
173
174            let xmm_a0 = Self::step(_mm_shuffle_epi8(self.fold[0], xmm_shl));
175
176            self.fold[0] = _mm_shuffle_epi8(self.fold[0], xmm_shr);
177            let xmm_tmp1 = _mm_shuffle_epi8(self.fold[1], xmm_shl);
178            self.fold[0] = _mm_or_si128(self.fold[0], xmm_tmp1);
179
180            self.fold[1] = _mm_shuffle_epi8(self.fold[1], xmm_shr);
181            let xmm_tmp2 = _mm_shuffle_epi8(self.fold[2], xmm_shl);
182            self.fold[1] = _mm_or_si128(self.fold[1], xmm_tmp2);
183
184            self.fold[2] = _mm_shuffle_epi8(self.fold[2], xmm_shr);
185            let xmm_tmp3 = _mm_shuffle_epi8(self.fold[3], xmm_shl);
186            self.fold[2] = _mm_or_si128(self.fold[2], xmm_tmp3);
187
188            self.fold[3] = _mm_shuffle_epi8(self.fold[3], xmm_shr);
189            let xmm_crc_part = _mm_shuffle_epi8(xmm_crc_part, xmm_shl);
190            self.fold[3] = _mm_or_si128(self.fold[3], xmm_crc_part);
191
192            // zlib-ng uses casts and a floating-point xor instruction here. There is a theory that
193            // this breaks dependency chains on some CPUs and gives better throughput. Other sources
194            // claim that casting between integer and float has a cost and should be avoided. We can't
195            // measure the difference, and choose the shorter code.
196            self.fold[3] = _mm_xor_si128(self.fold[3], xmm_a0)
197        }
198    }

200    #[allow(clippy::needless_range_loop)]
201    unsafe fn progress<const N: usize, const COPY: bool>(
202        &mut self,
203        dst: &mut [u8],
204        src: &mut &[u8],
205        init_crc: &mut u32,
206    ) -> usize {
207        let mut it = src.chunks_exact(16);
208        let mut input: [_; N] = core::array::from_fn(|_| unsafe {
209            _mm_load_si128(it.next().unwrap().as_ptr() as *const __m128i)
210        });
211
212        *src = &src[N * 16..];
213
214        if COPY {
215            for (s, d) in input[..N].iter().zip(dst.chunks_exact_mut(16)) {
216                unsafe { _mm_storeu_si128(d.as_mut_ptr() as *mut __m128i, *s) };
217            }
218        } else if *init_crc != CRC32_INITIAL_VALUE {
219            let xmm_initial = reg([*init_crc, 0, 0, 0]);
220            input[0] = unsafe { _mm_xor_si128(input[0], xmm_initial) };
221            *init_crc = CRC32_INITIAL_VALUE;
222        }
223
224        unsafe { self.fold_step::<N>() };
225
226        for i in 0..N {
227            self.fold[i + (4 - N)] = unsafe { _mm_xor_si128(self.fold[i + (4 - N)], input[i]) };
228        }
229
230        if COPY {
231            N * 16
232        } else {
233            0
234        }
235    }

237    #[target_feature(enable = "pclmulqdq", enable = "sse2", enable = "sse4.1")]
238    unsafe fn fold_help<const COPY: bool>(
239        &mut self,
240        mut dst: &mut [u8],
241        mut src: &[u8],
242        mut init_crc: u32,
243    ) {
244        let mut xmm_crc_part = reg([0; 4]);
245
246        let mut partial_buf = Align16([0u8; 16]);
247
248        // Technically the CRC functions don't even call this for input < 64, but a bare minimum of 31
249        // bytes of input is needed for the aligning load that occurs.  If there's an initial CRC, to
250        // carry it forward through the folded CRC there must be 16 - src % 16 + 16 bytes available, which
251        // by definition can be up to 15 bytes + one full vector load. */
252        assert!(src.len() >= 31 || init_crc == CRC32_INITIAL_VALUE);
253
254        if COPY {
255            assert_eq!(dst.len(), src.len(), "dst and src must be the same length")
256        }
257
258        if src.len() < 16 {
259            if COPY {
260                if src.is_empty() {
261                    return;
262                }
263
264                partial_buf.0[..src.len()].copy_from_slice(src);
265                xmm_crc_part =
266                    unsafe { _mm_load_si128(partial_buf.0.as_mut_ptr() as *mut __m128i) };
267                dst[..src.len()].copy_from_slice(&partial_buf.0[..src.len()]);
268            }
269        } else {
270            // SAFETY: [u8; 16] can safely transmute into _m128i.
271            let (before, _, _) = unsafe { src.align_to::<__m128i>() };
272
273            #[allow(unused_unsafe)] // because target features 1.1
274            if !before.is_empty() {
275                xmm_crc_part = unsafe { _mm_loadu_si128(src.as_ptr() as *const __m128i) };
276                if COPY {
277                    unsafe { _mm_storeu_si128(dst.as_mut_ptr() as *mut __m128i, xmm_crc_part) };
278                    dst = &mut dst[before.len()..];
279                } else {
280                    let is_initial = init_crc == CRC32_INITIAL_VALUE;
281
282                    if !is_initial {
283                        let xmm_initial = reg([init_crc, 0, 0, 0]);
284                        xmm_crc_part = unsafe { _mm_xor_si128(xmm_crc_part, xmm_initial) };
285                        init_crc = CRC32_INITIAL_VALUE;
286                    }
287
288                    if before.len() < 4 && !is_initial {
289                        let xmm_t0 = xmm_crc_part;
290                        xmm_crc_part =
291                            unsafe { _mm_loadu_si128((src.as_ptr() as *const __m128i).add(1)) };
292
293                        unsafe { self.fold_step::<1>() };
294
295                        self.fold[3] = unsafe { _mm_xor_si128(self.fold[3], xmm_t0) };
296                        src = &src[16..];
297                    }
298                }
299
300                unsafe { self.partial_fold(xmm_crc_part, before.len()) };
301
302                src = &src[before.len()..];
303            }
304
305            // if is_x86_feature_detected!("vpclmulqdq") {
306            //     if src.len() >= 256 {
307            //         if COPY {
308            //             // size_t n = fold_16_vpclmulqdq_copy(&xmm_crc0, &xmm_crc1, &xmm_crc2, &xmm_crc3, dst, src, len);
309            //             // dst += n;
310            //         } else {
311            //             // size_t n = fold_16_vpclmulqdq(&xmm_crc0, &xmm_crc1, &xmm_crc2, &xmm_crc3, src, len, xmm_initial, first);
312            //             // first = false;
313            //         }
314            //         // len -= n;
315            //         // src += n;
316            //     }
317            // }
318
319            while src.len() >= 64 {
320                let n = unsafe { self.progress::<4, COPY>(dst, &mut src, &mut init_crc) };
321                dst = &mut dst[n..];
322            }
323
324            if src.len() >= 48 {
325                let n = unsafe { self.progress::<3, COPY>(dst, &mut src, &mut init_crc) };
326                dst = &mut dst[n..];
327            } else if src.len() >= 32 {
328                let n = unsafe { self.progress::<2, COPY>(dst, &mut src, &mut init_crc) };
329                dst = &mut dst[n..];
330            } else if src.len() >= 16 {
331                let n = unsafe { self.progress::<1, COPY>(dst, &mut src, &mut init_crc) };
332                dst = &mut dst[n..];
333            }
334        }
335
336        if !src.is_empty() {
337            debug_assert!(src.len() <= 16);
338            unsafe {
339                // SAFETY: src and xmm_crc_part don't overlap. xmm_crc_part is statically
340                // guaranteed to be 16 bytes, and in earlier code we advanced the src slice such
341                // that it's less than 16 bytes.
342                core::ptr::copy_nonoverlapping(
343                    src.as_ptr(),
344                    &mut xmm_crc_part as *mut _ as *mut u8,
345                    src.len(),
346                );
347                if COPY {
348                    _mm_storeu_si128(partial_buf.0.as_mut_ptr() as *mut __m128i, xmm_crc_part);
349                    core::ptr::copy_nonoverlapping(
350                        partial_buf.0.as_ptr(),
351                        dst.as_mut_ptr(),
352                        src.len(),
353                    );
354                }
355                self.partial_fold(xmm_crc_part, src.len());
356            }
357        }
358    }
