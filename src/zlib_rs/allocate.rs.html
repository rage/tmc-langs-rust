20/// # Safety
21///
22/// This function is safe, but must have this type signature to be used elsewhere in the library
23#[cfg(unix)]
24unsafe extern "C" fn zalloc_c(opaque: *mut c_void, items: c_uint, size: c_uint) -> *mut c_void {
25    let _ = opaque;
26
27    extern "C" {
28        fn posix_memalign(memptr: *mut *mut c_void, align: size_t, size: size_t) -> c_int;
29    }
30
31    let mut ptr = core::ptr::null_mut();
32    match posix_memalign(&mut ptr, ALIGN.into(), items as size_t * size as size_t) {
33        0 => ptr,
34        _ => core::ptr::null_mut(),
35    }
36}

82/// # Safety
83///
84/// This function is safe to call.
85#[cfg(feature = "rust-allocator")]
86unsafe extern "C" fn zalloc_rust(_opaque: *mut c_void, count: c_uint, size: c_uint) -> *mut c_void {
87    let size = count as usize * size as usize;
88
89    // internally, we want to align allocations to 64 bytes (in part for SIMD reasons)
90    let layout = Layout::from_size_align(size, ALIGN.into()).unwrap();
91
92    let ptr = std::alloc::System.alloc(layout);
93
94    ptr as *mut c_void
95}

116/// # Safety
117///
118/// - `ptr` must be allocated with the rust `alloc::System` allocator
119/// - `opaque` is a `&usize` that represents the size of the allocation
120#[cfg(feature = "rust-allocator")]
121unsafe extern "C" fn zfree_rust(opaque: *mut c_void, ptr: *mut c_void) {
122    if ptr.is_null() {
123        return;
124    }
125
126    // we can't really do much else. Deallocating with an invalid layout is UB.
127    debug_assert!(!opaque.is_null());
128    if opaque.is_null() {
129        return;
130    }
131
132    let size = *(opaque as *mut usize);
133
134    let layout = Layout::from_size_align(size, ALIGN.into());
135    let layout = layout.unwrap();
136
137    std::alloc::System.dealloc(ptr.cast(), layout);
138}

185impl Allocator<'_> {
186    fn allocate_layout(&self, layout: Layout) -> *mut c_void {
187        assert!(layout.align() <= ALIGN.into());
188
189        // Special case for the Rust `alloc` backed allocator
190        #[cfg(feature = "rust-allocator")]
191        if self.zalloc == Allocator::RUST.zalloc {
192            let ptr = unsafe { (Allocator::RUST.zalloc)(self.opaque, layout.size() as _, 1) };
193
194            debug_assert_eq!(ptr as usize % layout.align(), 0);
195
196            return ptr;
197        }
198
199        // General case for c-style allocation
200
201        // We cannot rely on the allocator giving properly aligned allocations and have to fix that ourselves.
202        //
203        // The general approach is to allocate a bit more than the layout needs, so that we can
204        // give the application a properly aligned address and also store the real allocation
205        // pointer in the allocation so that `free` can free the real allocation pointer.
206        //
207        //
208        // Example: The layout represents `(u32, u32)`, with an alignment of 4 bytes and a
209        // total size of 8 bytes.
210        //
211        // Assume that the allocator will give us address `0x07`. We need that to be a multiple
212        // of the alignment, so that shifts the starting position to `0x08`. Then we also need
213        // to store the pointer to the start of the allocation so that `free` can free that
214        // pointer, bumping to `0x10`. The `0x10` pointer is then the pointer that the application
215        // deals with. When free'ing, the original allocation pointer can be read from `0x10 - size_of::<*const c_void>()`.
216        //
217        // Of course there does need to be enough space in the allocation such that when we
218        // shift the start forwards, the end is still within the allocation. Hence we allocate
219        // `extra_space` bytes: enough for a full alignment plus a pointer.
220
221        // we need at least
222        //
223        // - `align` extra space so that no matter what pointer we get from zalloc, we can shift the start of the
224        //      allocation by at most `align - 1` so that `ptr as usize % align == 0
225        // - `size_of::<*mut _>` extra space so that after aligning to `align`,
226        //      there is `size_of::<*mut _>` space to store the pointer to the allocation.
227        //      This pointer is then retrieved in `free`
228        let extra_space = core::mem::size_of::<*mut c_void>() + layout.align();
229
230        // Safety: we assume allocating works correctly in the safety assumptions on
231        // `DeflateStream` and `InflateStream`.
232        let ptr = unsafe { (self.zalloc)(self.opaque, (layout.size() + extra_space) as _, 1) };
233
234        if ptr.is_null() {
235            return ptr;
236        }
237
238        // Calculate return pointer address with space enough to store original pointer
239        let align_diff = (ptr as usize).next_multiple_of(layout.align()) - (ptr as usize);
240
241        // Safety: offset is smaller than 64, and we allocated 64 extra bytes in the allocation
242        let mut return_ptr = unsafe { ptr.cast::<u8>().add(align_diff) };
243
244        // if there is not enough space to store a pointer we need to make more
245        if align_diff < core::mem::size_of::<*mut c_void>() {
246            // # Safety
247            //
248            // - `return_ptr` is well-aligned, therefore `return_ptr + align` is also well-aligned
249            // - we reserve `size_of::<*mut _> + align` extra space in the allocation, so
250            //      `ptr + align_diff + align` is still valid for (at least) `layout.size` bytes
251            let offset = Ord::max(core::mem::size_of::<*mut c_void>(), layout.align());
252            return_ptr = unsafe { return_ptr.add(offset) };
253        }
254
255        // Store the original pointer for free()
256        //
257        // Safety: `align >= size_of::<*mut _>`, so there is now space for a pointer before `return_ptr`
258        // in the allocation
259        unsafe {
260            let original_ptr = return_ptr.sub(core::mem::size_of::<*mut c_void>());
261            core::ptr::write_unaligned(original_ptr.cast::<*mut c_void>(), ptr);
262        };
263
264        // Return properly aligned pointer in allocation
265        let ptr = return_ptr.cast::<c_void>();
266
267        debug_assert_eq!(ptr as usize % layout.align(), 0);
268
269        ptr
270    }
271
272    fn allocate_layout_zeroed(&self, layout: Layout) -> *mut c_void {
273        assert!(layout.align() <= ALIGN.into());
274
275        #[cfg(feature = "rust-allocator")]
276        if self.zalloc == Allocator::RUST.zalloc {
277            let ptr = unsafe { zalloc_rust_calloc(self.opaque, layout.size() as _, 1) };
278
279            debug_assert_eq!(ptr as usize % layout.align(), 0);
280
281            return ptr;
282        }
283
284        #[cfg(feature = "c-allocator")]
285        if self.zalloc == Allocator::C.zalloc {
286            let alloc = Allocator {
287                zalloc: zalloc_c_calloc,
288                zfree: zfree_c,
289                opaque: core::ptr::null_mut(),
290                _marker: PhantomData,
291            };
292
293            return alloc.allocate_layout(layout);
294        }
295
296        // create the allocation (contents are uninitialized)
297        let ptr = self.allocate_layout(layout);
298
299        if !ptr.is_null() {
300            // zero all contents (thus initializing the buffer)
301            unsafe { core::ptr::write_bytes(ptr, 0u8, layout.size()) };
302        }
303
304        ptr
305    }
306
307    pub fn allocate_raw<T>(&self) -> Option<NonNull<T>> {
308        NonNull::new(self.allocate_layout(Layout::new::<T>()).cast())
309    }
310
311    pub fn allocate_slice_raw<T>(&self, len: usize) -> Option<NonNull<T>> {
312        NonNull::new(self.allocate_layout(Layout::array::<T>(len).ok()?).cast())
313    }
314
315    pub fn allocate_zeroed_raw<T>(&self) -> Option<NonNull<T>> {
316        NonNull::new(self.allocate_layout_zeroed(Layout::new::<T>()).cast())
317    }
318
319    pub fn allocate_zeroed_buffer(&self, len: usize) -> Option<NonNull<u8>> {
320        let layout = Layout::array::<u8>(len).ok()?;
321        NonNull::new(self.allocate_layout_zeroed(layout).cast())
322    }
323
324    /// # Panics
325    ///
326    /// - when `len` is 0
327    ///
328    /// # Safety
329    ///
330    /// - `ptr` must be allocated with this allocator
331    /// - `len` must be the number of `T`s that are in this allocation
332    #[allow(unused)] // Rust needs `len` for deallocation
333    pub unsafe fn deallocate<T>(&self, ptr: *mut T, len: usize) {
334        if !ptr.is_null() {
335            // Special case for the Rust `alloc` backed allocator
336            #[cfg(feature = "rust-allocator")]
337            if self.zfree == Allocator::RUST.zfree {
338                assert_ne!(len, 0, "invalid size for {:?}", ptr);
339                let mut size = core::mem::size_of::<T>() * len;
340                return (Allocator::RUST.zfree)(&mut size as *mut usize as *mut c_void, ptr.cast());
341            }
342
343            // General case for c-style allocation
344            let original_ptr = (ptr as *mut u8).sub(core::mem::size_of::<*const c_void>());
345            let free_ptr = core::ptr::read_unaligned(original_ptr as *mut *mut c_void);
346
347            (self.zfree)(self.opaque, free_ptr)
348        }
349    }
