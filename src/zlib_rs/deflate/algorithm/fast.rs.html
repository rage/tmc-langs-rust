3use crate::deflate::hash_calc::StandardHashCalc;
4use crate::{
5    deflate::{
6        fill_window, BlockState, DeflateStream, MIN_LOOKAHEAD, STD_MIN_MATCH, WANT_MIN_MATCH,
7    },
8    flush_block, DeflateFlush,
9};
10
11pub fn deflate_fast(stream: &mut DeflateStream, flush: DeflateFlush) -> BlockState {
12    loop {
13        // Make sure that we always have enough lookahead, except
14        // at the end of the input file. We need STD_MAX_MATCH bytes
15        // for the next match, plus WANT_MIN_MATCH bytes to insert the
16        // string following the next match.
17        if stream.state.lookahead < MIN_LOOKAHEAD {
18            fill_window(stream);
19            if stream.state.lookahead < MIN_LOOKAHEAD && flush == DeflateFlush::NoFlush {
20                return BlockState::NeedMore;
21            }
22            if stream.state.lookahead == 0 {
23                break; /* flush the current block */
24            }
25        }
26
27        let state = &mut stream.state;
28
29        // Insert the string window[strstart .. strstart+2] in the
30        // dictionary, and set hash_head to the head of the hash chain:
31
32        let lc: u8; // Literal character to output if there is no match.
33        if state.lookahead >= WANT_MIN_MATCH {
34            let val = u32::from_le_bytes(
35                state.window.filled()[state.strstart..state.strstart + 4]
36                    .try_into()
37                    .unwrap(),
38            );
39            let hash_head = StandardHashCalc::quick_insert_value(state, state.strstart, val);
40            let dist = state.strstart as isize - hash_head as isize;
41
42            // Find the longest match for the string starting at offset state.strstart.
43            if dist <= state.max_dist() as isize && dist > 0 && hash_head != 0 {
44                // To simplify the code, we prevent matches with the string
45                // of window index 0 (in particular we have to avoid a match
46                // of the string with itself at the start of the input file).
47                let mut match_len;
48                (match_len, state.match_start) =
49                    crate::deflate::longest_match::longest_match(state, hash_head);
50                if match_len >= WANT_MIN_MATCH {
51                    let bflush = state.tally_dist(
52                        state.strstart - state.match_start as usize,
53                        match_len - STD_MIN_MATCH,
54                    );
55
56                    state.lookahead -= match_len;
57
58                    /* Insert new strings in the hash table only if the match length
59                     * is not too large. This saves time but degrades compression.
60                     */
61                    if match_len <= state.max_insert_length() && state.lookahead >= WANT_MIN_MATCH {
62                        match_len -= 1; /* string at strstart already in table */
63                        state.strstart += 1;
64
65                        state.insert_string(state.strstart, match_len);
66                        state.strstart += match_len;
67                    } else {
68                        state.strstart += match_len;
69                        StandardHashCalc::quick_insert_string(
70                            state,
71                            state.strstart + 2 - STD_MIN_MATCH,
72                        );
73
74                        /* If lookahead < STD_MIN_MATCH, ins_h is garbage, but it does not
75                         * matter since it will be recomputed at next deflate call.
76                         */
77                    }
78                    if bflush {
79                        flush_block!(stream, false);
80                    }
81                    continue;
82                }
83            }
84            lc = val as u8;
85        } else {
86            lc = state.window.filled()[state.strstart];
87        }
88        /* No match, output a literal byte */
89        let bflush = state.tally_lit(lc);
90        state.lookahead -= 1;
91        state.strstart += 1;
92        if bflush {
93            flush_block!(stream, false);
94        }
95    }
96
97    stream.state.insert = if stream.state.strstart < (STD_MIN_MATCH - 1) {
98        stream.state.strstart
99    } else {
100        STD_MIN_MATCH - 1
101    };
102
103    if flush == DeflateFlush::Finish {
104        flush_block!(stream, true);
105        return BlockState::FinishDone;
106    }
107
108    if !stream.state.sym_buf.is_empty() {
109        flush_block!(stream, false);
110    }
111
112    BlockState::BlockDone
113}
