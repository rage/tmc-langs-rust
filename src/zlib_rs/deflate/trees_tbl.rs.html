73/// Precomputes the `Values` generated by the `encode_len` function for all `u8` input values
74#[rustfmt::skip]
75pub const STATIC_LTREE_ENCODINGS: [Value; 256] = {
76    let mut table = [Value::new(0, 0); 256];
77
78    let mut lc = 0;
79    while lc < table.len() {
80        let (code, len) = super::encode_len(&STATIC_LTREE, lc as u8);
81
82        // assert that there is no precision loss
83        assert!(code as u16 as u64 == code);
84        assert!(len as u16 as usize == len);
85
86        table[lc] = Value::new(code as u16, len as u16);
87
88        lc += 1;
89    }
90
91    table
92};
