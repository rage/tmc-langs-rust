3#[derive(Debug, Clone, Copy)]
4pub enum HashCalcVariant {
5    Standard,
6    Roll,
7}
8
9impl HashCalcVariant {
10    /// Use rolling hash for deflate_slow algorithm with level 9. It allows us to
11    /// properly lookup different hash chains to speed up longest_match search.
12    pub fn for_max_chain_length(max_chain_length: u16) -> Self {
13        if max_chain_length > 1024 {
14            HashCalcVariant::Roll
15        } else {
16            HashCalcVariant::Standard
17        }
18    }
19}

21pub struct StandardHashCalc;
22
23impl StandardHashCalc {
24    const HASH_CALC_OFFSET: usize = 0;
25
26    const HASH_CALC_MASK: u32 = (HASH_SIZE - 1) as u32;
27
28    fn hash_calc(_: u32, val: u32) -> u32 {
29        const HASH_SLIDE: u32 = 16;
30        val.wrapping_mul(2654435761) >> HASH_SLIDE
31    }
32
33    pub fn update_hash(h: u32, val: u32) -> u32 {
34        Self::hash_calc(h, val) & Self::HASH_CALC_MASK
35    }
36
37    #[inline]
38    pub fn quick_insert_string(state: &mut State, string: usize) -> u16 {
39        let slice = &state.window.filled()[string + Self::HASH_CALC_OFFSET..];
40        let val = u32::from_le_bytes(slice[..4].try_into().unwrap());
41
42        Self::quick_insert_value(state, string, val)
43    }
44
45    #[inline]
46    pub fn quick_insert_value(state: &mut State, string: usize, val: u32) -> u16 {
47        let hm = Self::update_hash(0, val) as usize;
48
49        let head = state.head.as_slice()[hm];
50        if head != string as u16 {
51            state.prev.as_mut_slice()[string & state.w_mask()] = head;
52            state.head.as_mut_slice()[hm] = string as u16;
53        }
54
55        head
56    }
57
58    pub fn insert_string(state: &mut State, string: usize, count: usize) {
59        let slice = &state.window.filled()[string + Self::HASH_CALC_OFFSET..];
60
61        // it can happen that insufficient bytes are initialized
62        // .take(count) generates worse assembly
63        let slice = &slice[..Ord::min(slice.len(), count + 3)];
64
65        let w_mask = state.w_mask();
66        for (i, w) in slice.windows(4).enumerate() {
67            let idx = string as u16 + i as u16;
68
69            let val = u32::from_le_bytes(w.try_into().unwrap());
70
71            let hm = Self::update_hash(0, val) as usize;
72
73            let head = state.head.as_slice()[hm];
74            if head != idx {
75                state.prev.as_mut_slice()[idx as usize & w_mask] = head;
76                state.head.as_mut_slice()[hm] = idx;
77            }
78        }
79    }
80}

82pub struct RollHashCalc;
83
84impl RollHashCalc {
85    const HASH_CALC_OFFSET: usize = STD_MIN_MATCH - 1;
86
87    const HASH_CALC_MASK: u32 = (1 << 15) - 1;
88
89    fn hash_calc(h: u32, val: u32) -> u32 {
90        const HASH_SLIDE: u32 = 5;
91        (h << HASH_SLIDE) ^ val
92    }
93
94    pub fn update_hash(h: u32, val: u32) -> u32 {
95        Self::hash_calc(h, val) & Self::HASH_CALC_MASK
96    }
97
98    pub fn quick_insert_string(state: &mut State, string: usize) -> u16 {
99        let val = state.window.filled()[string + Self::HASH_CALC_OFFSET] as u32;
100
101        state.ins_h = Self::hash_calc(state.ins_h, val);
102        state.ins_h &= Self::HASH_CALC_MASK;
103
104        let hm = state.ins_h as usize;
105
106        let head = state.head.as_slice()[hm];
107        if head != string as u16 {
108            state.prev.as_mut_slice()[string & state.w_mask()] = head;
109            state.head.as_mut_slice()[hm] = string as u16;
110        }
111
112        head
113    }
114
115    pub fn insert_string(state: &mut State, string: usize, count: usize) {
116        let slice = &state.window.filled()[string + Self::HASH_CALC_OFFSET..][..count];
117
118        let w_mask = state.w_mask();
119        for (i, val) in slice.iter().copied().enumerate() {
120            let idx = string as u16 + i as u16;
121
122            state.ins_h = Self::hash_calc(state.ins_h, val as u32);
123            state.ins_h &= Self::HASH_CALC_MASK;
124            let hm = state.ins_h as usize;
125
126            let head = state.head.as_slice()[hm];
127            if head != idx {
128                state.prev.as_mut_slice()[idx as usize & w_mask] = head;
129                state.head.as_mut_slice()[hm] = idx;
130            }
131        }
132    }
133}
