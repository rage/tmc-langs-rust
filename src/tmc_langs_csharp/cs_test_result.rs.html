1//! Contains the CSTestResult type that models the C# test runner result.
2
3use serde::Deserialize;
4use std::collections::HashSet;
5use tmc_langs_framework::TestResult;
6
7/// Test result from the C# test runner.
8#[derive(Debug, Deserialize)]
9#[serde(rename_all = "PascalCase")]
10#[allow(clippy::upper_case_acronyms)]
11pub struct CSTestResult {
12    pub name: String,
13    pub passed: bool,
14    pub message: String,
15    pub points: Vec<String>,
16    pub error_stack_trace: Vec<String>,
17}
18
19impl CSTestResult {
20    pub fn into_test_result(mut self, failed_points: &HashSet<String>) -> TestResult {
21        self.points.retain(|point| !failed_points.contains(point));
22        TestResult {
23            name: self.name,
24            successful: self.passed,
25            message: self.message,
26            exception: self.error_stack_trace,
27            points: self.points,
28        }
29    }
30}
