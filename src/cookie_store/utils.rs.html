35pub fn is_secure(url: &Url) -> bool {
36    if url.scheme() == "https" {
37        return true;
38    }
39    if let Some(u) = url.host() {
40        match u {
41            Host::Domain(d) => d == "localhost",
42            Host::Ipv4(ip) => ip.is_loopback(),
43            Host::Ipv6(ip) => ip.is_loopback(),
44        }
45    } else {
46        false
47    }
48}
