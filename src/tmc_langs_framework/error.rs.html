67// == Collection of errors likely to be useful in multiple plugins which can be special cased without needing a plugin's specific error type ==
68/// An error caused by a failed attempt to execute an external command.
69#[derive(Error, Debug)]
70pub enum CommandError {
71    #[error("Failed to execute command: {0}")]
72    Popen(String, #[source] PopenError),
73    #[error(
74        "The executable for command {cmd} could not be found. Please make sure you have installed it correctly."
75    )]
76    NotFound { cmd: String, source: PopenError },
77    #[error("Failed to run command {0}")]
78    FailedToRun(String, #[source] PopenError),
79    #[error("Command {command} exited with status {status:?}")]
80    Failed {
81        command: String,
82        status: ExitStatus,
83        stdout: String,
84        stderr: String,
85    },
86    #[error("Command {command} timed out after {} seconds.", .timeout.as_secs())]
87    TimeOut {
88        command: String,
89        timeout: Duration,
90        stdout: String,
91        stderr: String,
92    },
93    #[error("Failed to terminate command {0}")]
94    Terminate(String, #[source] std::io::Error),
95}
