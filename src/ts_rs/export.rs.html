16/// An error which may occur when exporting a type
17#[derive(Error, Debug)]
18pub enum ExportError {
19    #[error("this type cannot be exported")]
20    CannotBeExported(&'static str),
21    #[cfg(feature = "format")]
22    #[error("an error occurred while formatting the generated typescript output")]
23    Formatting(String),
24    #[error("an error occurred while performing IO")]
25    Io(#[from] std::io::Error),
26    #[error("the environment variable CARGO_MANIFEST_DIR is not set")]
27    ManifestDirNotSet,
28}

31mod recursive_export {
32    use std::{any::TypeId, collections::HashSet};
33
34    use super::export_type;
35    use crate::{
36        typelist::{TypeList, TypeVisitor},
37        ExportError, TS,
38    };
39
40    struct Visit<'a> {
41        seen: &'a mut HashSet<TypeId>,
42        error: Option<ExportError>,
43    }
44
45    impl<'a> TypeVisitor for Visit<'a> {
46        fn visit<T: TS + 'static + ?Sized>(&mut self) {
47            // if an error occurred previously, or the type cannot be exported (it's a primitive),
48            // we return
49            if self.error.is_some() || T::EXPORT_TO.is_none() {
50                return;
51            }
52
53            self.error = export_recursive::<T>(self.seen).err();
54        }
55    }
56
57    /// Exports `T` to the file specified by the `#[ts(export_to = ..)]` attribute.
58    /// Additionally, all dependencies of `T` will be exported as well.
59    ///
60    /// TODO: This might cause a race condition:
61    ///       If two types `A` and `B` are `#[ts(export)]` and depend on type `C`,
62    ///       then both tests for exporting `A` and `B` will try to write `C` to `C.ts`.
63    ///       Since rust, by default, executes tests in paralell, this might cause `C.ts` to be corrupted.
64    pub(crate) fn export_type_with_dependencies<T: TS + ?Sized + 'static>(
65    ) -> Result<(), ExportError> {
66        let mut seen = HashSet::new();
67        export_recursive::<T>(&mut seen)
68    }
69
70    // exports T, then recursively calls itself with all of its dependencies
71    fn export_recursive<T: TS + ?Sized + 'static>(
72        seen: &mut HashSet<TypeId>,
73    ) -> Result<(), ExportError> {
74        if !seen.insert(TypeId::of::<T>()) {
75            return Ok(());
76        }
77
78        export_type::<T>()?;
79
80        let mut visitor = Visit { seen, error: None };
81        T::dependency_types().for_each(&mut visitor);
82
83        if let Some(e) = visitor.error {
84            Err(e)
85        } else {
86            Ok(())
87        }
88    }
89}

91/// Export `T` to the file specified by the `#[ts(export_to = ..)]` attribute
92pub(crate) fn export_type<T: TS + ?Sized + 'static>() -> Result<(), ExportError> {
93    let path = output_path::<T>()?;
94    export_type_to::<T, _>(&path)
95}

97/// Export `T` to the file specified by the `path` argument.
98pub(crate) fn export_type_to<T: TS + ?Sized + 'static, P: AsRef<Path>>(
99    path: P,
100) -> Result<(), ExportError> {
101    // Lock to make sure only one file will be written at a time.
102    // In the future, it might make sense to replace this with something more clever to only prevent
103    // two threads from writing the **same** file concurrently.
104    static FILE_LOCK: Mutex<()> = Mutex::new(());
105
106    #[allow(unused_mut)]
107    let mut buffer = export_type_to_string::<T>()?;
108
109    // format output
110    #[cfg(feature = "format")]
111    {
112        use dprint_plugin_typescript::{configuration::ConfigurationBuilder, format_text};
113
114        let fmt_cfg = ConfigurationBuilder::new().deno().build();
115        if let Some(formatted) =
116            format_text(path.as_ref(), &buffer, &fmt_cfg).map_err(|e| Formatting(e.to_string()))?
117        {
118            buffer = formatted;
119        }
120    }
121
122    if let Some(parent) = path.as_ref().parent() {
123        std::fs::create_dir_all(parent)?;
124    }
125    let lock = FILE_LOCK.lock().unwrap();
126    std::fs::write(path.as_ref(), buffer)?;
127    drop(lock);
128    Ok(())
129}

135    const EXPORT_DIR_ENV_VAR: &str = "TS_RS_EXPORT_DIR";
136    fn provided_default_dir() -> Option<&'static str> {
137        static EXPORT_TO: OnceLock<Option<String>> = OnceLock::new();
138        EXPORT_TO
139            .get_or_init(|| std::env::var(EXPORT_DIR_ENV_VAR).ok())
140            .as_deref()
141    }
142
143    /// Returns the path to where `T` should be exported using the `TS_RS_EXPORT_DIR` environment variable.
144    ///
145    /// This should only be used by the TS derive macro; the `get_export_to` trait method should not
146    /// be overridden if the `#[ts(export_to = ..)]` attribute exists.
147    pub fn get_export_to_path<T: TS + ?Sized>() -> Option<String> {
148        provided_default_dir().map_or_else(
149            || T::EXPORT_TO.map(ToString::to_string),
150            |path| Some(format!("{path}/{}.ts", T::name())),
151        )
152    }
153}

155/// Returns the generated defintion for `T`.
156pub(crate) fn export_type_to_string<T: TS + ?Sized + 'static>() -> Result<String, ExportError> {
157    let mut buffer = String::with_capacity(1024);
158    buffer.push_str(NOTE);
159    generate_imports::<T>(&mut buffer)?;
160    generate_decl::<T>(&mut buffer);
161    Ok(buffer)
162}

164/// Compute the output path to where `T` should be exported.
165fn output_path<T: TS + ?Sized>() -> Result<PathBuf, ExportError> {
166    let manifest_dir = std::env::var("CARGO_MANIFEST_DIR").map_err(|_| ManifestDirNotSet)?;
167    let manifest_dir = Path::new(&manifest_dir);
168    let path =
169        PathBuf::from(T::get_export_to().ok_or(CannotBeExported(std::any::type_name::<T>()))?);
170    Ok(manifest_dir.join(path))
171}

173/// Push the declaration of `T`
174fn generate_decl<T: TS + ?Sized>(out: &mut String) {
175    // Type Docs
176    let docs = &T::DOCS;
177    if let Some(docs) = docs {
178        out.push_str(docs);
179    }
180
181    // Type Definition
182    out.push_str("export ");
183    out.push_str(&T::decl());
184}

186/// Push an import statement for all dependencies of `T`
187fn generate_imports<T: TS + ?Sized + 'static>(out: &mut String) -> Result<(), ExportError> {
188    let export_to = T::get_export_to().ok_or(CannotBeExported(std::any::type_name::<T>()))?;
189    let path = Path::new(&export_to);
190
191    let deps = T::dependencies();
192    let deduplicated_deps = deps
193        .iter()
194        .filter(|dep| dep.type_id != TypeId::of::<T>())
195        .map(|dep| (&dep.ts_name, dep))
196        .collect::<BTreeMap<_, _>>();
197
198    for (_, dep) in deduplicated_deps {
199        let rel_path = import_path(path, Path::new(&dep.exported_to));
200        writeln!(
201            out,
202            "import type {{ {} }} from {:?};",
203            &dep.ts_name, rel_path
204        )
205        .unwrap();
206    }
207    writeln!(out).unwrap();
208    Ok(())
209}

211/// Returns the required import path for importing `import` from the file `from`
212fn import_path(from: &Path, import: &Path) -> String {
213    let rel_path =
214        diff_paths(import, from.parent().unwrap()).expect("failed to calculate import path");
215    let path = match rel_path.components().next() {
216        Some(Component::Normal(_)) => format!("./{}", rel_path.to_string_lossy()),
217        _ => rel_path.to_string_lossy().into(),
218    };
219
220    let path_without_extension = path.trim_end_matches(".ts");
221
222    if cfg!(feature = "import-esm") {
223        format!("{}.js", path_without_extension)
224    } else {
225        path_without_extension.to_owned()
226    }
227}

234// http://www.apache.org/licenses/LICENSE-2.0> or the MIT license
235// <LICENSE-MIT or http://opensource.org/licenses/MIT>, at your

237// except according to those terms.
238//
239// Adapted from rustc's path_relative_from
240// https://github.com/rust-lang/rust/blob/e1d0de82cc40b666b88d4a6d2c9dcbc81d7ed27f/src/librustc_back/rpath.rs#L116-L158
241fn diff_paths<P, B>(path: P, base: B) -> Option<PathBuf>
242where
243    P: AsRef<Path>,
244    B: AsRef<Path>,
245{
246    let path = path.as_ref();
247    let base = base.as_ref();
248
249    if path.is_absolute() != base.is_absolute() {
250        if path.is_absolute() {
251            Some(PathBuf::from(path))
252        } else {
253            None
254        }
255    } else {
256        let mut ita = path.components();
257        let mut itb = base.components();
258        let mut comps: Vec<Component> = vec![];
259        loop {
260            match (ita.next(), itb.next()) {
261                (None, None) => break,
262                (Some(a), None) => {
263                    comps.push(a);
264                    comps.extend(ita.by_ref());
265                    break;
266                }
267                (None, _) => comps.push(Component::ParentDir),
268                (Some(a), Some(b)) if comps.is_empty() && a == b => (),
269                (Some(a), Some(Component::CurDir)) => comps.push(a),
270                (Some(_), Some(Component::ParentDir)) => return None,
271                (Some(a), Some(_)) => {
272                    comps.push(Component::ParentDir);
273                    for _ in itb {
274                        comps.push(Component::ParentDir);
275                    }
276                    comps.push(a);
277                    comps.extend(ita.by_ref());
278                    break;
279                }
280            }
281        }
282        Some(comps.iter().map(|c| c.as_os_str()).collect())
283    }
