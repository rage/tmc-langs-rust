1//! Serialize a Rust data structure into a `JsValue`

3use crate::error::{Error, Result as LibResult};

8pub fn to_value<'js, C, V>(cx: &mut C, value: &V) -> LibResult<Handle<'js, JsValue>>
9where
10    C: Context<'js>,
11    V: Serialize + ?Sized,
12{
13    let serializer = Serializer {
14        cx,
15        ph: PhantomData,
16    };
17    let serialized_value = value.serialize(serializer)?;
18    Ok(serialized_value)
19}

137    fn serialize_char(self, v: char) -> Result<Self::Ok, Self::Error> {
138        let mut b = [0; 4];
139        let result = v.encode_utf8(&mut b);
140        let js_str =
141            JsString::try_new(self.cx, result).map_err(|_| Error::StringTooLongForChar(4))?;
142        Ok(js_str.upcast())
143    }
144
145    fn serialize_str(self, v: &str) -> Result<Self::Ok, Self::Error> {
146        let len = v.len();
147        let js_str = JsString::try_new(self.cx, v).map_err(|_| Error::StringTooLong(len))?;
148        Ok(js_str.upcast())
149    }

151    fn serialize_bytes(self, v: &[u8]) -> Result<Self::Ok, Self::Error> {
152        let mut buff = JsBuffer::new(self.cx, v.len())?;
153        buff.as_mut_slice(self.cx).clone_from_slice(v);
154        Ok(buff.upcast())
155    }

272impl<'j, C> ser::SerializeSeq for ArraySerializer<'_, 'j, C>
273where
274    C: Context<'j>,
275{
276    type Ok = Handle<'j, JsValue>;
277    type Error = Error;
278
279    fn serialize_element<T>(&mut self, value: &T) -> Result<(), Self::Error>
280    where
281        T: Serialize + ?Sized,
282    {
283        let value = to_value(self.cx, value)?;
284
285        let arr: Handle<'j, JsArray> = self.array;
286        let len = arr.len(self.cx);
287        arr.set(self.cx, len, value)?;
288        Ok(())
289    }
290
291    fn end(self) -> Result<Self::Ok, Self::Error> {
292        Ok(self.array.upcast())
293    }
294}

387impl<'j, C> ser::SerializeMap for MapSerializer<'_, 'j, C>
388where
389    C: Context<'j>,
390{
391    type Ok = Handle<'j, JsValue>;
392    type Error = Error;
393
394    fn serialize_key<T>(&mut self, key: &T) -> Result<(), Self::Error>
395    where
396        T: Serialize + ?Sized,
397    {
398        let key = to_value(self.cx, key)?;
399        self.key_holder.set(self.cx, "key", key)?;
400        Ok(())
401    }
402
403    fn serialize_value<T>(&mut self, value: &T) -> Result<(), Self::Error>
404    where
405        T: Serialize + ?Sized,
406    {
407        let key: Handle<'j, JsValue> = self.key_holder.get(&mut *self.cx, "key")?;
408        let value_obj = to_value(self.cx, value)?;
409        self.object.set(self.cx, key, value_obj)?;
410        Ok(())
411    }
412
413    fn end(self) -> Result<Self::Ok, Self::Error> {
414        Ok(self.object.upcast())
415    }
416}
