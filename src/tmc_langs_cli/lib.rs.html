35pub enum ParsingResult {
36    Ok(Cli),
37    Help(clap::Error),
38    Version(clap::Error),
39    Err(CliOutput),
40}
41
42pub fn map_parsing_result(result: Result<Cli, clap::Error>) -> ParsingResult {
43    match result {
44        Ok(cli) => ParsingResult::Ok(cli),
45        Err(e) if e.kind() == clap::error::ErrorKind::DisplayHelp => ParsingResult::Help(e),
46        Err(e) if e.kind() == clap::error::ErrorKind::DisplayVersion => ParsingResult::Version(e),
47        Err(e) => {
48            // CLI was called incorrectly
49            let e = anyhow::Error::from(e).context("Failed to parse arguments");
50            let causes: Vec<String> = e.chain().map(|e| format!("Caused by: {e}")).collect();
51            let output = CliOutput::OutputData(Box::new(OutputData {
52                status: Status::Finished,
53                message: format!("{e:?}"), // debug formatting to print backtrace from anyhow
54                result: OutputResult::Error,
55                data: Some(DataKind::Error {
56                    kind: Kind::Generic,
57                    trace: causes,
58                }),
59            }));
60            ParsingResult::Err(output)
61        }
62    }
63}

71pub fn run(cli: Cli) -> Result<CliOutput, CliError> {
72    match run_app(cli) {
73        Ok(output) => Ok(output),
74        Err(e) => {
75            // error handling
76            let causes: Vec<String> = e.chain().map(|e| format!("Caused by: {e}")).collect();
77            let message = error_message_special_casing(&e);
78            let kind = solve_error_kind(&e);
79            let sandbox_path = check_sandbox_err(&e);
80            let output = CliOutput::OutputData(Box::new(OutputData {
81                status: Status::Finished,
82                message,
83                result: OutputResult::Error,
84                data: Some(DataKind::Error {
85                    kind,
86                    trace: causes,
87                }),
88            }));
89            Err(CliError {
90                output: Box::new(output),
91                sandbox_path,
92            })
93        }
94    }
95}

97/// Goes through the error chain and checks for special error types that should be indicated by the Kind.
98fn solve_error_kind(e: &anyhow::Error) -> Kind {
99    for cause in e.chain() {
100        // check for invalid token
101        if cause.downcast_ref::<InvalidTokenError>().is_some() {
102            return Kind::InvalidToken;
103        }
104
105        // check for tmc client errors
106        match cause.downcast_ref::<TestMyCodeClientError>() {
107            Some(TestMyCodeClientError::HttpError {
108                url: _,
109                status,
110                error: _,
111                obsolete_client,
112            }) => {
113                if *obsolete_client {
114                    return Kind::ObsoleteClient;
115                }
116                if status.as_u16() == 403 {
117                    return Kind::Forbidden;
118                }
119                if status.as_u16() == 401 {
120                    return Kind::NotLoggedIn;
121                }
122            }
123            Some(TestMyCodeClientError::NotAuthenticated) => {
124                return Kind::NotLoggedIn;
125            }
126            Some(TestMyCodeClientError::ConnectionError { .. }) => {
127                return Kind::ConnectionError;
128            }
129            _ => {}
130        }
131
132        // check for tmc client errors
133        match cause.downcast_ref::<MoocClientError>() {
134            Some(MoocClientError::HttpError {
135                url: _,
136                status,
137                error: _,
138                obsolete_client,
139            }) => {
140                if *obsolete_client {
141                    return Kind::ObsoleteClient;
142                }
143                if status.as_u16() == 403 {
144                    return Kind::Forbidden;
145                }
146                if status.as_u16() == 401 {
147                    return Kind::NotLoggedIn;
148                }
149            }
150            Some(MoocClientError::NotAuthenticated) => {
151                return Kind::NotLoggedIn;
152            }
153            Some(MoocClientError::ConnectionError { .. }) => {
154                return Kind::ConnectionError;
155            }
156            _ => {}
157        }
158
159        // check for download failed error
160        if let Some(DownloadsFailedError {
161            downloaded: completed,
162            skipped,
163            failed,
164        }) = cause.downcast_ref::<DownloadsFailedError>()
165        {
166            return Kind::FailedExerciseDownload {
167                completed: completed.clone(),
168                skipped: skipped.clone(),
169                failed: failed.clone(),
170            };
171        }
172    }
173
174    Kind::Generic
175}

177/// Goes through the error chain and returns the specialized error message, if any.
178fn error_message_special_casing(e: &anyhow::Error) -> String {
179    for cause in e.chain() {
180        // command not found errors are special cased to notify the user that they may need to install additional software
181        if let Some(cnf) = cause.downcast_ref::<CommandError>() {
182            return cnf.to_string();
183        }
184    }
185    e.to_string()
186}

188/// Goes through the error chain and returns the error output file path if a sandbox test error is found
189fn check_sandbox_err(e: &anyhow::Error) -> Option<PathBuf> {
190    for cause in e.chain() {
191        if let Some(SandboxTestError {
192            path: Some(path), ..
193        }) = cause.downcast_ref::<SandboxTestError>()
194        {
195            return Some(path.clone());
196        }
197    }
198    None
199}

522                    .expect("locale is required if checkstyle output path is given")
523                    .0;

525                run_checkstyle_write_results(
526                    &exercise_path,
527                    Some(&checkstyle_output_path),
528                    locale,
529                )?;
530            }
531
532            CliOutput::finished_with_data(
533                format!("ran tests for {}", exercise_path.display()),
534                DataKind::TestResult(test_result),
535            )
536        }
537
538        Command::Settings(settings) => run_settings(settings)?,
539
540        Command::ScanExercise {
541            exercise_path,
542            output_path,
543        } => {
544            let mut lock = Lock::dir(&exercise_path, LockOptions::Read)?;

573    Ok(output)

576fn run_tmc(tmc: TestMyCode) -> Result<CliOutput> {
577    let client_name = &tmc.client_name;
578    let client_version = &tmc.client_version;
579    let root_url = env::var("TMC_LANGS_TMC_ROOT_URL")
580        .unwrap_or_else(|_| "https://tmc.mooc.fi/".to_string())
581        .parse()
582        .context("Invalid TMC root url")?;
583    let (mut client, mut credentials) =
584        tmc_langs::init_testmycode_client_with_credentials(root_url, client_name, client_version)?;
585
586    match run_tmc_inner(tmc, &mut client, &mut credentials) {
587        Err(error) => {
588            for cause in error.chain() {
589                // check if the token was rejected and delete it if so
590                if let Some(TestMyCodeClientError::HttpError { status, .. }) =
591                    cause.downcast_ref::<TestMyCodeClientError>()
592                {
593                    if status.as_u16() == 401 {
594                        log::error!("Received HTTP 401 error, deleting credentials");
595                        if let Some(credentials) = credentials {
596                            credentials.remove()?;
597                        }
598                        return Err(InvalidTokenError { source: error }.into());
599                    }
600                }
601            }
602            Err(error)
603        }
604        output => output,
605    }
606}

1034fn run_mooc(mooc: Mooc) -> Result<CliOutput> {
1035    let root_url = env::var("TMC_LANGS_MOOC_ROOT_URL")
1036        .unwrap_or_else(|_| "https://courses.mooc.fi/".to_string())
1037        .parse()
1038        .context("Invalid TMC root url")?;
1039
1040    let (mut client, credentials) =
1041        tmc_langs::init_mooc_client_with_credentials(root_url, &mooc.client_name)?;
1042    match run_mooc_inner(mooc, &mut client) {
1043        Err(error) => {
1044            for cause in error.chain() {
1045                // check if the token was rejected and delete it if so
1046                if let Some(TestMyCodeClientError::HttpError { status, .. }) =
1047                    cause.downcast_ref::<TestMyCodeClientError>()
1048                {
1049                    if status.as_u16() == 401 {
1050                        log::error!("Received HTTP 401 error, deleting credentials");
1051                        if let Some(credentials) = credentials {
1052                            credentials.remove()?;
1053                        }
1054                        return Err(InvalidTokenError { source: error }.into());
1055                    } else {
1056                        log::warn!("401 without credentials");
1057                    }
1058                }
1059            }
1060            Err(error)
1061        }
1062        output => output,
1063    }
1064}

1202fn write_result_to_file_as_json<T: Serialize>(
1203    result: &T,
1204    output_path: &Path,
1205    pretty: bool,
1206    secret: Option<String>,
1207) -> Result<()> {
1208    let mut output_lock =
1209        Lock::file(output_path, LockOptions::WriteTruncate).with_context(|| {
1210            format!(
1211                "Failed to create results JSON file at {}",
1212                output_path.display()
1213            )
1214        })?;
1215    let mut output_guard = output_lock.lock()?;
1216
1217    if let Some(secret) = secret {
1218        let token = tmc_langs::sign_with_jwt(result, secret.as_bytes())?;
1219        file_util::write_to_writer(token, output_guard.get_file_mut())
1220            .with_context(|| format!("Failed to write result to {}", output_path.display()))?;
1221    } else if pretty {
1222        serde_json::to_writer_pretty(output_guard.get_file_mut(), result).with_context(|| {
1223            format!(
1224                "Failed to write result as JSON to {}",
1225                output_path.display()
1226            )
1227        })?;
1228    } else {
1229        serde_json::to_writer(output_guard.get_file_mut(), result).with_context(|| {
1230            format!(
1231                "Failed to write result as JSON to {}",
1232                output_path.display()
1233            )
1234        })?;
1235    }
1236
1237    Ok(())
1238}

1264                "Failed to write code style check results as JSON to {}",

1266            )
1267        })?;
