1//! Contains a struct representing valgrind's log output.

3use crate::error::MakeError;
4use once_cell::sync::Lazy;
5use regex::Regex;
6use std::{
7    collections::HashMap,
8    io::{BufRead, BufReader},
9    path::Path,
10};
11use tmc_langs_util::{FileError, file_util};
12
13#[derive(Debug)]
14#[allow(dead_code)]
15pub struct ValgrindLog {
16    #[allow(dead_code)]
17    pub header: (String, Vec<String>),
18    pub errors: bool,
19    pub results: Vec<ValgrindResult>,
20}
21
22impl ValgrindLog {
23    /// Attempts to read and parse the log file at the given path.
24    pub fn from(valgrind_log_path: &Path) -> Result<Self, MakeError> {
25        // TODO: use parsing lib?
26        log::debug!("parsing {}", valgrind_log_path.display());
27
28        #[allow(clippy::unwrap_used)]
29        static PID_REGEX: Lazy<Regex> = Lazy::new(|| Regex::new(r#"==(?P<pid>\d+)=="#).unwrap());
30        #[allow(clippy::unwrap_used)]
31        static ERR_REGEX: Lazy<Regex> =
32            Lazy::new(|| Regex::new(r#"== ERROR SUMMARY: (?P<error_count>\d+)"#).unwrap());
33
34        let valgrind_log_file = file_util::open_file(valgrind_log_path)?;
35        let valgrind_log = BufReader::new(valgrind_log_file);
36
37        let mut first_pid = None;
38        let mut pid_info = HashMap::new();
39        // parse all lines into a map of pid => ([lines of text], error count)
40        for line in valgrind_log.lines() {
41            let line = line.map_err(|e| FileError::FileRead(valgrind_log_path.to_path_buf(), e))?;
42            let pid = match PID_REGEX.captures(&line) {
43                Some(captures) => captures["pid"].to_string(),
44                None => continue, // ignore lines without a PID
45            };
46
47            first_pid.get_or_insert(pid.clone());
48            let info = pid_info.entry(pid).or_insert((vec![], 0));
49            if let Some(captures) = ERR_REGEX.captures(&line) {
50                let errors = captures["error_count"].parse::<u32>()?;
51                info.1 = errors;
52            }
53            info.0.push(line);
54        }
55
56        let first_pid = match first_pid {
57            Some(first_pid) => first_pid,
58            None => return Err(MakeError::NoPidsInValgrindLogs),
59        };
60        let (header_log, _header_errors) = pid_info
61            .remove(&first_pid)
62            .expect("pid_info should have info for every pid");
63
64        let mut contains_errors = false;
65        let mut results = vec![];
66        for (pid, (log, errors)) in pid_info {
67            let errors = errors > 0;
68            contains_errors = contains_errors || errors;
69            results.push(ValgrindResult { pid, errors, log })
70        }
71
72        let log = ValgrindLog {
73            header: (first_pid, header_log),
74            errors: contains_errors,
75            results,
76        };
77
78        log::trace!("parsed {log:#?}");
79        Ok(log)
80    }
81}

85pub struct ValgrindResult {
86    #[allow(dead_code)]
87    pub pid: String,
88    pub errors: bool,
89    pub log: Vec<String>,
90}

92#[cfg(test)]
93#[allow(clippy::unwrap_used)]
94mod test {
95    use super::*;
96
97    fn init() {
98        use log::*;
99        use simple_logger::*;
100        let _ = SimpleLogger::new().with_level(LevelFilter::Debug).init();
101    }
102
103    fn file_to(
104        target_dir: impl AsRef<std::path::Path>,
105        target_relative: impl AsRef<std::path::Path>,
106        contents: impl AsRef<[u8]>,
107    ) -> std::path::PathBuf {
108        let target = target_dir.as_ref().join(target_relative);
109        if let Some(parent) = target.parent() {
110            std::fs::create_dir_all(parent).unwrap();
111        }
112        std::fs::write(&target, contents.as_ref()).unwrap();
113        target
114    }
115
116    #[test]
117    fn parses_errors() {
118        init();
119
120        let temp_dir = tempfile::tempdir().unwrap();
121        let file = file_to(
122            &temp_dir,
123            "file",
124            r#"
125==1234== 
126==1234== stuff
127==1234== 
128==1234== ERROR SUMMARY: 11
129
130==2345==
131==2345== stuff
132==2345==
133==2345== ERROR SUMMARY: 22
134"#,
135        );
136
137        let valgrind_log = ValgrindLog::from(&file).unwrap();
138        log::debug!("{valgrind_log:#?}");

140    }
