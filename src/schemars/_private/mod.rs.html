16pub fn json_schema_for_internally_tagged_enum_newtype_variant<T: ?Sized + JsonSchema>(
17    generator: &mut SchemaGenerator,
18) -> Schema {
19    let mut schema = T::json_schema(generator);
20
21    // Inline the newtype's inner schema if any of:
22    // - The type specifies that its schema should always be inlined
23    // - The generator settings specify that all schemas should be inlined
24    // - The inner type is a unit struct, which would cause an unsatisfiable schema due to mismatched `type`.
25    //    In this case, we replace its type with "object" in `apply_internal_enum_variant_tag`
26    // - The inner schema specified `"additionalProperties": false` or `"unevaluatedProperties": false`,
27    //    since that would disallow the variant tag. If additional/unevaluatedProperties is in the top-level
28    //    schema, then we can leave it there, because it will "see" the variant tag property. But if it is
29    //    nested e.g. in an `allOf`, then it must be removed, which is why we run `AllowUnknownProperties`
30    //    but only on immediate subschemas.
31
32    let mut transform = AllowUnknownProperties::default();
33    transform_immediate_subschemas(&mut transform, &mut schema);
34
35    if T::inline_schema()
36        || generator.settings().inline_subschemas
37        || schema.get("type").and_then(Value::as_str) == Some("null")
38        || schema.get("additionalProperties").and_then(Value::as_bool) == Some(false)
39        || schema.get("unevaluatedProperties").and_then(Value::as_bool) == Some(false)
40        || transform.did_modify
41    {
42        return schema;
43    }
44
45    // ...otherwise, we can freely refer to the schema via a `$ref`
46    generator.subschema_for::<T>()
47}

49// Helper for generating schemas for flattened `Option` fields.
50pub fn json_schema_for_flatten<T: ?Sized + JsonSchema>(
51    generator: &mut SchemaGenerator,
52    required: bool,
53) -> Schema {
54    let mut schema = T::_schemars_private_non_optional_json_schema(generator);
55
56    if T::_schemars_private_is_option() && !required {
57        schema.remove("required");
58    }
59
60    // Always allow aditional/unevaluated properties, because the outer struct determines
61    // whether it denies unknown fields.
62    AllowUnknownProperties::default().transform(&mut schema);
63
64    schema
65}
66
67#[derive(Default)]
68struct AllowUnknownProperties {
69    did_modify: bool,
70}
71
72impl Transform for AllowUnknownProperties {
73    fn transform(&mut self, schema: &mut Schema) {
74        if schema.get("additionalProperties").and_then(Value::as_bool) == Some(false) {
75            schema.remove("additionalProperties");
76            self.did_modify = true;
77        }
78        if schema.get("unevaluatedProperties").and_then(Value::as_bool) == Some(false) {
79            schema.remove("unevaluatedProperties");
80            self.did_modify = true;
81        }
82
83        transform_immediate_subschemas(self, schema);
84    }
85}

174/// Update a schema for an internally tagged enum variant
175pub fn apply_internal_enum_variant_tag(
176    schema: &mut Schema,
177    tag_name: &str,
178    variant: &str,
179    deny_unknown_fields: bool,
180) {
181    let obj = schema.ensure_object();
182    let is_unit = obj.get("type").and_then(Value::as_str) == Some("null");
183
184    obj.insert("type".to_owned(), "object".into());
185
186    if let Some(properties) = obj
187        .entry("properties")
188        .or_insert(Value::Object(Map::new()))
189        .as_object_mut()
190    {
191        properties.insert(
192            tag_name.to_string(),
193            json!({
194                "type": "string",
195                "const": variant
196            }),
197        );
198    }
199
200    if let Some(required) = obj
201        .entry("required")
202        .or_insert(Value::Array(Vec::new()))
203        .as_array_mut()
204    {
205        required.insert(0, tag_name.into());
206    }
207
208    if deny_unknown_fields && is_unit {
209        obj.entry("additionalProperties").or_insert(false.into());
210    }
211}

213pub fn insert_object_property(
214    schema: &mut Schema,
215    key: &str,
216    is_optional: bool,
217    sub_schema: Schema,
218) {
219    let obj = schema.ensure_object();
220    if let Some(properties) = obj
221        .entry("properties")
222        .or_insert(Value::Object(Map::new()))
223        .as_object_mut()
224    {
225        properties.insert(key.to_owned(), sub_schema.into());
226    }
227
228    if !is_optional {
229        if let Some(req) = obj
230            .entry("required")
231            .or_insert(Value::Array(Vec::new()))
232            .as_array_mut()
233        {
234            req.push(key.into());
235        }
236    }

272pub fn flatten(schema: &mut Schema, other: Schema) {
273    fn flatten_property(obj1: &mut Map<String, Value>, key: String, value2: Value) {
274        match obj1.entry(key) {
275            Entry::Vacant(vacant) => {
276                vacant.insert(value2);
277            }
278            Entry::Occupied(occupied) => {
279                match occupied.key().as_str() {
280                    "required" | "allOf" => {
281                        if let Value::Array(a1) = occupied.into_mut() {
282                            if let Value::Array(a2) = value2 {
283                                a1.extend(a2);
284                            }
285                        }
286                    }
287                    "properties" | "patternProperties" => {
288                        if let Value::Object(o1) = occupied.into_mut() {
289                            if let Value::Object(o2) = value2 {
290                                o1.extend(o2);
291                            }
292                        }
293                    }
294                    "oneOf" | "anyOf" => {
295                        let (key, current) = occupied.remove_entry();
296                        flatten_property(
297                            obj1,
298                            "allOf".to_owned(),
299                            json!([
300                                { &key: current },
301                                { key: value2 }
302                            ]),
303                        );
304                    }
305                    _ => {
306                        // leave the original value as it is (don't modify `schema`)
307                    }
308                }
309            }
310        }
311    }
312
313    match other.try_to_object() {
314        Err(false) => {}
315        Err(true) => {
316            if let Some(obj) = schema.as_object_mut() {
317                if !obj.contains_key("additionalProperties")
318                    && !obj.contains_key("unevaluatedProperties")
319                {
320                    let key = if contains_immediate_subschema(obj) {
321                        "unevaluatedProperties"
322                    } else {
323                        "additionalProperties"
324                    };
325                    obj.insert(key.to_owned(), true.into());
326                }
327            }
328        }
329        Ok(mut obj2) => {
330            let obj1 = schema.ensure_object();
331
332            // For complex merges, replace `additionalProperties` with `unevaluatedProperties`
333            // which usually "works out better".
334            normalise_additional_unevaluated_properties(obj1, &obj2);
335            normalise_additional_unevaluated_properties(&mut obj2, obj1);
336
337            for (key, value2) in obj2 {
338                flatten_property(obj1, key, value2);
339            }
340        }
341    }
342}
343
344fn normalise_additional_unevaluated_properties(
345    schema_obj1: &mut Map<String, Value>,
346    schema_obj2: &Map<String, Value>,
347) {
348    if schema_obj1.contains_key("additionalProperties")
349        && (schema_obj2.contains_key("unevaluatedProperties")
350            || contains_immediate_subschema(schema_obj2))
351    {
352        let ap = schema_obj1.remove("additionalProperties");
353        schema_obj1.insert("unevaluatedProperties".to_owned(), ap.into());
354    }
355}

357fn contains_immediate_subschema(schema_obj: &Map<String, Value>) -> bool {
358    ["if", "allOf", "anyOf", "oneOf", "$ref"]
359        .into_iter()
360        .any(|k| schema_obj.contains_key(k))
361}

363pub(crate) fn allow_null(generator: &mut SchemaGenerator, schema: &mut Schema) {
364    fn is_null_schema(value: &Value) -> bool {
365        <&Schema>::try_from(value).is_ok_and(|s| s.has_type("null"))
366    }
367
368    match schema.try_as_object_mut() {
369        Ok(obj) => {
370            if obj.len() == 1
371                && obj
372                    .get("anyOf")
373                    .and_then(Value::as_array)
374                    .is_some_and(|a| a.iter().any(is_null_schema))
375            {
376                return;
377            }
378
379            if contains_immediate_subschema(obj) {
380                *schema = json_schema!({
381                    "anyOf": [
382                        obj,
383                        <()>::json_schema(generator)
384                    ]
385                });
386                // No need to check `type`/`const`/`enum` because they're trivially not present
387                return;
388            }
389
390            if let Some(instance_type) = obj.get_mut("type") {
391                match instance_type {
392                    Value::Array(array) => {
393                        let null = Value::from("null");
394                        if !array.contains(&null) {
395                            array.push(null);
396                        }
397                    }
398                    Value::String(string) => {
399                        if string != "null" {
400                            let current_type = core::mem::take(string).into();
401                            *instance_type = Value::Array(vec![current_type, "null".into()]);
402                        }
403                    }
404                    _ => {}
405                }
406            }
407
408            if let Some(c) = obj.remove("const") {
409                if !c.is_null() {
410                    obj.insert("enum".to_string(), Value::Array(vec![c, Value::Null]));
411                }
412            } else if let Some(Value::Array(e)) = obj.get_mut("enum") {
413                if !e.contains(&Value::Null) {
414                    e.push(Value::Null);
415                }
416            }
417        }
418        Err(true) => {}
419        Err(false) => {
420            *schema = <()>::json_schema(generator);
421        }
422    }
423}
